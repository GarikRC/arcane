100     /**
101      * Add a curve to the specified location (specifing the number of segments)
102      *
103      * @param x        The destination x coordinate
104      * @param y        The destination y coordiante
105      * @param cx1      The x coordiante of the first control point
106      * @param cy1      The y coordiante of the first control point
107      * @param cx2      The x coordinate of the second control point
108      * @param cy2      The y coordinate of the second control point
109      * @param segments The number of segments to use for the new curve
110      */
111     public void curveTo(float x, float y, float cx1, float cy1, float cx2, float cy2, int segments) {
112         // special case for zero movement
113         if ((cx == x) && (cy == y)) {
114             return;
115         }
116 
117         Curve curve = new Curve(new Vector2f(cx, cy), new Vector2f(cx1, cy1), new Vector2f(cx2, cy2), new Vector2f(x, y));
118         float step = 1.0f / segments;
119 
120         for (int i = 1; i < segments + 1; i++) {
121             float t = i * step;
122             Vector2f p = curve.pointAt(t);
123             if (hole != null) {
124                 hole.add(new float[]{p.x, p.y});
125             } else {
126                 localPoints.add(new float[]{p.x, p.y});
127             }
128             cx = p.x;
129             cy = p.y;
130         }
131         pointsDirty = true;
132     }

137     protected void createPoints() {
138         points = new float[localPoints.size() * 2];
139         for (int i = 0; i < localPoints.size(); i++) {
140             float[] p = (float[]) localPoints.get(i);
141             points[(i * 2)] = p[0];
142             points[(i * 2) + 1] = p[1];
143         }
144     }

146     /**
147      * @see org.newdawn.slick.geom.Shape#transform(org.newdawn.slick.geom.Transform)
148      */
149     public Shape transform(Transform transform) {
150         Path p = new Path(cx, cy);
151         p.localPoints = transform(localPoints, transform);
152         for (int i = 0; i < holes.size(); i++) {
153             p.holes.add(transform((ArrayList) holes.get(i), transform));
154         }
155         p.closed = this.closed;
156 
157         return p;
158     }

160     /**
161      * Transform a list of points
162      *
163      * @param pts The pts to transform
164      * @param t   The transform to apply
165      * @return The transformed points
166      */
167     private ArrayList transform(ArrayList pts, Transform t) {
168         float[] in = new float[pts.size() * 2];
169         float[] out = new float[pts.size() * 2];
170 
171         for (int i = 0; i < pts.size(); i++) {
172             in[i * 2] = ((float[]) pts.get(i))[0];
173             in[(i * 2) + 1] = ((float[]) pts.get(i))[1];
174         }
175         t.transform(in, 0, out, 0, pts.size());
176 
177         ArrayList outList = new ArrayList();
178         for (int i = 0; i < pts.size(); i++) {
179             outList.add(new float[]{out[(i * 2)], out[(i * 2) + 1]});
180         }
181 
182         return outList;
