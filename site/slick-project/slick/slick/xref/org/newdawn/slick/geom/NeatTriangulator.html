74      /**
75       * Find an edge between two verts
76       *
77       * @param i The index of the first vert
78       * @param j The index of the second vert
79       * @return The index of the dge
80       */
81      private int findEdge(int i, int j) {
82          int k;
83          int l;
84          if (i < j) {
85              k = i;
86              l = j;
87          } else {
88              k = j;
89              l = i;
90          }
91          for (int i1 = 0; i1 < numEdges; i1++)
92              if (edges[i1].v0 == k && edges[i1].v1 == l)
93                  return i1;
94  
95          return -1;
96      }

98      /**
99       * Add a discovered edge
100      *
101      * @param i The index of the first vert
102      * @param j The index of the second vert
103      * @param k The index of the spread vert
104      */
105     private void addEdge(int i, int j, int k) {
106         int l1 = findEdge(i, j);
107         int j1;
108         int k1;
109         Edge edge;
110         if (l1 < 0) {
111             if (numEdges == edges.length) {
112                 Edge aedge[] = new Edge[edges.length * 2];
113                 System.arraycopy(edges, 0, aedge, 0, numEdges);
114                 edges = aedge;
115             }
116             j1 = -1;
117             k1 = -1;
118             l1 = numEdges++;
119             edge = edges[l1] = new Edge();
120         } else {
121             edge = edges[l1];
122             j1 = edge.t0;
123             k1 = edge.t1;
124         }
125         int l;
126         int i1;
127         if (i < j) {
128             l = i;
129             i1 = j;
130             j1 = k;
131         } else {
132             l = j;
133             i1 = i;
134             k1 = k;
135         }
136         edge.v0 = l;
137         edge.v1 = i1;
138         edge.t0 = j1;
139         edge.t1 = k1;
140         edge.suspect = true;
141     }

178     /**
179      * Choose the suspect to become part of the triangle
180      *
181      * @return The edge selected
182      */
183     private Edge chooseSuspect() {
184         for (int i = 0; i < numEdges; i++) {
185             Edge edge = edges[i];
186             if (edge.suspect) {
187                 edge.suspect = false;
188                 if (edge.t0 >= 0 && edge.t1 >= 0)
189                     return edge;
190             }
191         }
192 
193         return null;
194     }

196     /**
197      * Factor rho.
198      *
199      * @param f  Factor 1
200      * @param f1 Factor 2
201      * @param f2 Factor 3
202      * @param f3 Factor 4
203      * @param f4 Factor 5
204      * @param f5 Factor 6
205      * @return The computation of rho
206      */
207     private static float rho(float f, float f1, float f2, float f3, float f4, float f5) {
208         float f6 = f4 - f2;
209         float f7 = f5 - f3;
210         float f8 = f - f4;
211         float f9 = f1 - f5;
212         float f18 = f6 * f9 - f7 * f8;
213         if (f18 > 0.0F) {
214             if (f18 < 1E-006F)
215                 f18 = 1E-006F;
216             float f12 = f6 * f6;
217             float f13 = f7 * f7;
218             float f14 = f8 * f8;
219             float f15 = f9 * f9;
220             float f10 = f2 - f;
221             float f11 = f3 - f1;
222             float f16 = f10 * f10;
223             float f17 = f11 * f11;
224             return ((f12 + f13) * (f14 + f15) * (f16 + f17)) / (f18 * f18);
225         } else {
226             return -1F;
227         }
228     }

230     /**
231      * Check if the point P is inside the triangle defined by
232      * the points A,B,C
233      *
234      * @param f  Point A x-coordinate
235      * @param f1 Point A y-coordinate
236      * @param f2 Point B x-coordinate
237      * @param f3 Point B y-coordinate
238      * @param f4 Point C x-coordinate
239      * @param f5 Point C y-coordinate
240      * @param f6 Point P x-coordinate
241      * @param f7 Point P y-coordinate
242      * @return True if the point specified is within the triangle
243      */
244     private static boolean insideTriangle(float f, float f1, float f2, float f3, float f4, float f5, float f6, float f7) {
245         float f8 = f4 - f2;
246         float f9 = f5 - f3;
247         float f10 = f - f4;
248         float f11 = f1 - f5;
249         float f12 = f2 - f;
250         float f13 = f3 - f1;
251         float f14 = f6 - f;
252         float f15 = f7 - f1;
253         float f16 = f6 - f2;
254         float f17 = f7 - f3;
255         float f18 = f6 - f4;
256         float f19 = f7 - f5;
257         float f22 = f8 * f17 - f9 * f16;
258         float f20 = f12 * f15 - f13 * f14;
259         float f21 = f10 * f19 - f11 * f18;
260         return f22 >= 0.0D && f21 >= 0.0D && f20 >= 0.0D;
261     }

263     /**
264      * Cut a the contour and add a triangle into V to describe the
265      * location of the cut
266      *
267      * @param i The index of the first point
268      * @param j The index of the second point
269      * @param k The index of the third point
270      * @param l ?
271      * @return True if a triangle was found
272      */
273     private boolean snip(int i, int j, int k, int l) {
274         float f = pointsX[V[i]];
275         float f1 = pointsY[V[i]];
276         float f2 = pointsX[V[j]];
277         float f3 = pointsY[V[j]];
278         float f4 = pointsX[V[k]];
279         float f5 = pointsY[V[k]];
280         if (1E-006F > (f2 - f) * (f5 - f1) - (f3 - f1) * (f4 - f))
281             return false;
282         for (int i1 = 0; i1 < l; i1++)
283             if (i1 != i && i1 != j && i1 != k) {
284                 float f6 = pointsX[V[i1]];
285                 float f7 = pointsY[V[i1]];
286                 if (insideTriangle(f, f1, f2, f3, f4, f5, f6, f7))
287                     return false;
288             }
289 
290         return true;
291     }

293     /**
294      * Get the area defined by the points
295      *
296      * @return The area defined by the points
297      */
298     private float area() {
299         float f = 0.0F;
300         int i = numPoints - 1;
301         for (int j = 0; j < numPoints; ) {
302             f += pointsX[i] * pointsY[j] - pointsY[i] * pointsX[j];
303             i = j++;
304         }
305 
306         return f * 0.5F;
307     }

309     /**
310      * Perform simple triangulation
311      *
312      * @throws InternalException Indicates a polygon that can't be triangulated
313      */
314     public void basicTriangulation() throws InternalException {
315         int i = numPoints;
316         if (i < 3)
317             return;
318         numEdges = 0;
319         numTriangles = 0;
320         V = new int[i];
321 
322         if (0.0D < area()) {
323             for (int k = 0; k < i; k++)
324                 V[k] = k;
325 
326         } else {
327             for (int l = 0; l < i; l++)
328                 V[l] = numPoints - 1 - l;
329 
330         }
331         int k1 = 2 * i;
332         int i1 = i - 1;
333         while (i > 2) {
334             if (0 >= k1--) {
335                 throw new InternalException("Bad polygon");
336             }
337 
338             int j = i1;
339             if (i <= j)
340                 j = 0;
341             i1 = j + 1;
342             if (i <= i1)
343                 i1 = 0;
344             int j1 = i1 + 1;
345             if (i <= j1)
346                 j1 = 0;
347             if (snip(j, i1, j1, i)) {
348                 int l1 = V[j];
349                 int i2 = V[i1];
350                 int j2 = V[j1];
351                 if (numTriangles == triangles.length) {
352                     Triangle atriangle[] = new Triangle[triangles.length * 2];
353                     System.arraycopy(triangles, 0, atriangle, 0, numTriangles);
354                     triangles = atriangle;
355                 }
356                 triangles[numTriangles] = new Triangle(l1, i2, j2);
357                 addEdge(l1, i2, numTriangles);
358                 addEdge(i2, j2, numTriangles);
359                 addEdge(j2, l1, numTriangles);
360                 numTriangles++;
361                 int k2 = i1;
362                 for (int l2 = i1 + 1; l2 < i; l2++) {
363                     V[k2] = V[l2];
364                     k2++;
365                 }
366 
367                 i--;
368                 k1 = 2 * i;
369             }
370         }
371         V = null;
372     }

374     /**
375      * Optimize the triangulation by applying delauney rules
376      *
377      * @throws InternalException Indicates an invalid polygon
378      */
379     private void optimize() throws InternalException {
380         do {
381             Edge edge;
382             if ((edge = chooseSuspect()) == null) {
383                 break;
384             }
385             int i1 = edge.v0;
386             int k1 = edge.v1;
387             int i = edge.t0;
388             int j = edge.t1;
389             int j1 = -1;
390             int l1 = -1;
391             for (int k = 0; k < 3; k++) {
392                 int i2 = triangles[i].v[k];
393                 if (i1 == i2 || k1 == i2) {
394                     continue;
395                 }
396                 l1 = i2;
397                 break;
398             }
399 
400             for (int l = 0; l < 3; l++) {
401                 int j2 = triangles[j].v[l];
402                 if (i1 == j2 || k1 == j2) {
403                     continue;
404                 }
405                 j1 = j2;
406                 break;
407             }
408 
409             if (-1 == j1 || -1 == l1) {
410                 throw new InternalException("can't find quad");
411             }
412 
413             float f = pointsX[i1];
414             float f1 = pointsY[i1];
415             float f2 = pointsX[j1];
416             float f3 = pointsY[j1];
417             float f4 = pointsX[k1];
418             float f5 = pointsY[k1];
419             float f6 = pointsX[l1];
420             float f7 = pointsY[l1];
421             float f8 = rho(f, f1, f2, f3, f4, f5);
422             float f9 = rho(f, f1, f4, f5, f6, f7);
423             float f10 = rho(f2, f3, f4, f5, f6, f7);
424             float f11 = rho(f2, f3, f6, f7, f, f1);
425             if (0.0F > f8 || 0.0F > f9) {
426                 throw new InternalException("original triangles backwards");
427             }
428             if (0.0F <= f10 && 0.0F <= f11) {
429                 if (f8 > f9) {
430                     f8 = f9;
431                 }
432                 if (f10 > f11) {
433                     f10 = f11;
434                 }
435                 if (f8 > f10) {
436                     deleteEdge(i1, k1);
437                     triangles[i].v[0] = j1;
438                     triangles[i].v[1] = k1;
439                     triangles[i].v[2] = l1;
440                     triangles[j].v[0] = j1;
441                     triangles[j].v[1] = l1;
442                     triangles[j].v[2] = i1;
443                     addEdge(j1, k1, i);
444                     addEdge(k1, l1, i);
445                     addEdge(l1, j1, i);
446                     addEdge(l1, i1, j);
447                     addEdge(i1, j1, j);
448                     addEdge(j1, l1, j);
449                     markSuspect(j1, l1, false);
450                 }
451             }
452         } while (true);
453     }

469     /**
470      * Add a point to the polygon
471      */
472     public void addPolyPoint(float x, float y) {
473         for (int i = 0; i < numPoints; i++) {
474             if ((pointsX[i] == x) && (pointsY[i] == y)) {
475                 //return;
476                 y += offset;
477                 offset += EPSILON;
478             }
479         }
480 
481         if (numPoints == pointsX.length) {
482             float af[] = new float[numPoints * 2];
483             System.arraycopy(pointsX, 0, af, 0, numPoints);
484             pointsX = af;
485             af = new float[numPoints * 2];
486             System.arraycopy(pointsY, 0, af, 0, numPoints);
487             pointsY = af;
488         }
489 
490         pointsX[numPoints] = x;
491         pointsY[numPoints] = y;
492         numPoints++;
493     }
