40      /**
41       * Creates an image intended for use with offscreen rendering. Only one
42       * texture is created (the FBO/PBuffer-bound texture which will be used
43       * internally). This replaces the old way of offscreen rendering, using
44       * <tt>new Image(width, height)</tt>.
45       *
46       * @param width  the width of the offscreen image
47       * @param height the height of the offscreen image
48       * @param filter the desired filtering (FILTER_NEAREST or FILTER_LINEAR)
49       * @return a new Image prepared for use with getGraphics()
50       * @throws SlickException if there was a problem constructing the offscreen image
51       */
52      public static Image createOffscreenImage(int width, int height, int filter) throws SlickException {
53          // this is a bit hackish; ideally FBO/Image should be restructured into
54          // a more OpenGL-like design...
55          // but that would introduce a major overhaul of the library
56          Image i = new Image();
57          i.width = width;
58          i.height = height;
59          i.filter = filter;
60          i.inited = true; // so that initImpl() only gets called once
61          i.getGraphics(); // will call Image.setTexture, which calls reinit
62          return i;
63      }

507     /**
508      * Clamp the loaded texture to it's edges
509      */
510     public void clampTexture() {
511         if (GL.canTextureMirrorClamp()) {
512             GL.glTexParameteri(SGL.GL_TEXTURE_2D, SGL.GL_TEXTURE_WRAP_S, SGL.GL_MIRROR_CLAMP_TO_EDGE_EXT);
513             GL.glTexParameteri(SGL.GL_TEXTURE_2D, SGL.GL_TEXTURE_WRAP_T, SGL.GL_MIRROR_CLAMP_TO_EDGE_EXT);
514         } else {
515             GL.glTexParameteri(SGL.GL_TEXTURE_2D, SGL.GL_TEXTURE_WRAP_S, SGL.GL_CLAMP);
516             GL.glTexParameteri(SGL.GL_TEXTURE_2D, SGL.GL_TEXTURE_WRAP_T, SGL.GL_CLAMP);
517         }
518     }

665     /**
666      * Unlike the other drawEmbedded methods, this allows for the embedded image
667      * to be rotated. This is done by applying a rotation transform to each
668      * vertex of the image. This ignores getRotation but depends on the
669      * center x/y (scaled accordingly to the new width/height).
670      *
671      * @param x        the x to render the image at
672      * @param y        the y to render the image at
673      * @param width    the new width to render the image
674      * @param height   the new height to render the image
675      * @param rotation the rotation to render the image in degrees, using getCenterOfRotationX/Y
676      */
677     public void drawEmbedded(float x, float y, float width, float height, float rotation) {
678         if (rotation == 0) {
679             drawEmbedded(x, y, width, height);
680             return;
681         }
682         init();
683         float scaleX = width / this.width;
684         float scaleY = height / this.height;
685 
686         float cx = getCenterOfRotationX() * scaleX;
687         float cy = getCenterOfRotationY() * scaleY;
688 
689         float p1x = -cx;
690         float p1y = -cy;
691         float p2x = width - cx;
692         float p2y = -cy;
693         float p3x = width - cx;
694         float p3y = height - cy;
695         float p4x = -cx;
696         float p4y = height - cy;
697 
698         double rad = Math.toRadians(rotation);
699         final float cos = (float) FastTrig.cos(rad);
700         final float sin = (float) FastTrig.sin(rad);
701 
702         float tx = getTextureOffsetX();
703         float ty = getTextureOffsetY();
704         float tw = getTextureWidth();
705         float th = getTextureHeight();
706 
707         float x1 = (cos * p1x - sin * p1y) + cx; // TOP LEFT
708         float y1 = (sin * p1x + cos * p1y) + cy;
709         float x2 = (cos * p4x - sin * p4y) + cx; // BOTTOM LEFT
710         float y2 = (sin * p4x + cos * p4y) + cy;
711         float x3 = (cos * p3x - sin * p3y) + cx; // BOTTOM RIGHT
712         float y3 = (sin * p3x + cos * p3y) + cy;
713         float x4 = (cos * p2x - sin * p2y) + cx; // TOP RIGHT
714         float y4 = (sin * p2x + cos * p2y) + cy;
715         if (corners == null) {
716             GL.glTexCoord2f(tx, ty);
717             GL.glVertex3f(x + x1, y + y1, 0);
718             GL.glTexCoord2f(tx, ty + tw);
719             GL.glVertex3f(x + x2, y + y2, 0);
720             GL.glTexCoord2f(tx + tw, ty + th);
721             GL.glVertex3f(x + x3, y + y3, 0);
722             GL.glTexCoord2f(tx + tw, ty);
723             GL.glVertex3f(x + x4, y + y4, 0);
724         } else {
725             corners[TOP_LEFT].bind();
