29      /**
30       * Static initialiser for the codes created against Base64
31       */
32      static {
33          for (int i = 0; i < 256; i++)
34              baseCodes[i] = -1;
35          for (int i = 'A'; i <= 'Z'; i++)
36              baseCodes[i] = (byte) (i - 'A');
37          for (int i = 'a'; i <= 'z'; i++)
38              baseCodes[i] = (byte) (26 + i - 'a');
39          for (int i = '0'; i <= '9'; i++)
40              baseCodes[i] = (byte) (52 + i - '0');
41          baseCodes['+'] = 62;
42          baseCodes['/'] = 63;
43      }

89      /**
90       * Create a new layer based on the XML definition
91       *
92       * @param element The XML element describing the layer
93       * @param map     The map this layer is part of
94       * @throws SlickException Indicates a failure to parse the XML layer
95       */
96      public Layer(TiledMap map, Element element) throws SlickException {
97          this.map = map;
98          if (map instanceof TiledMapPlus) {
99              tmap = (TiledMapPlus) map;
100         }
101         name = element.getAttribute("name");
102         width = Integer.parseInt(element.getAttribute("width"));
103         height = Integer.parseInt(element.getAttribute("height"));
104         data = new int[width][height][3];
105         String opacityS = element.getAttribute("opacity");
106         if (!opacityS.equals("")) {
107             opacity = Float.parseFloat(opacityS);
108         }
109         if (element.getAttribute("visible").equals("0")) {
110             visible = false;
111         }
112 
113         // now read the layer properties
114         Element propsElement = (Element) element.getElementsByTagName(
115                 "properties").item(0);
116         if (propsElement != null) {
117             NodeList properties = propsElement.getElementsByTagName("property");
118             if (properties != null) {
119                 props = new Properties();
120                 for (int p = 0; p < properties.getLength(); p++) {
121                     Element propElement = (Element) properties.item(p);
122 
123                     String name = propElement.getAttribute("name");
124                     String value = propElement.getAttribute("value");
125                     props.setProperty(name, value);
126                 }
127             }
128         }
129 
130         Element dataNode = (Element) element.getElementsByTagName("data").item(
131                 0);
132         String encoding = dataNode.getAttribute("encoding");
133         String compression = dataNode.getAttribute("compression");
134 
135         if (encoding.equals("base64") && compression.equals("gzip")) {
136             try {
137                 Node cdata = dataNode.getFirstChild();
138                 char[] enc = cdata.getNodeValue().trim().toCharArray();
139                 byte[] dec = decodeBase64(enc);
140                 GZIPInputStream is = new GZIPInputStream(
141                         new ByteArrayInputStream(dec));
142                 readData(is);
143             } catch (IOException e) {
144                 Log.error(e);
145                 throw new SlickException("Unable to decode base 64 block");
146             }
147         } else if (encoding.equals("base64") && compression.equals("zlib")) {
148             Node cdata = dataNode.getFirstChild();
149             char[] enc = cdata.getNodeValue().trim().toCharArray();
150             byte[] dec = decodeBase64(enc);
151             InflaterInputStream is = new InflaterInputStream(
152                     new ByteArrayInputStream(dec));
153             readData(is);
154         } else {
155             throw new SlickException("Unsupport tiled map type: " + encoding
156                     + "," + compression + " (only gzip/zlib base64 supported)");
157         }
158     }

160     /**
161      * For reading decompressed, decoded Layer data into this layer
162      *
163      * @param is
164      */
165     protected void readData(InputStream is) {
166         for (int y = 0; y < height; y++) {
167             for (int x = 0; x < width; x++) {
168                 int tileId = 0;
169                 try {
170                     tileId |= is.read();
171                     tileId |= is.read() << 8;
172                     tileId |= is.read() << 16;
173                     tileId |= is.read() << 24;
174                 } catch (Exception e) {
175                     e.printStackTrace();
176                 }
177                 if (tileId == 0) {
178                     data[x][y][0] = -1;
179                     data[x][y][1] = 0;
180                     data[x][y][2] = 0;
181                 } else {
182                     int realTileId = tileId & 0x1FFFFFFF;
183                     TileSet set = map.findTileSet(realTileId);
184                     if (set != null) {
185                         data[x][y][0] = set.index;
186                         data[x][y][1] = realTileId - set.firstGID;
187                     }
188                     data[x][y][2] = tileId;
189                 }
190             }
191         }
192     }

216             data[x][y][2] = 0;
217         } else {

219 
220             data[x][y][0] = set.index; // tileSetIndex
221             data[x][y][1] = tile - set.firstGID; // localID
222             data[x][y][2] = tile; // globalID

226     /**
227      * Render a section of this layer
228      *
229      * @param x             The x location to render at
230      * @param y             The y location to render at
231      * @param sx            The x tile location to start rendering
232      * @param sy            The y tile location to start rendering
233      * @param width         The number of tiles across to render
234      * @param ty            The line of tiles to render
235      * @param lineByLine    True if we should render line by line, i.e. giving us a chance
236      *                      to render something else between lines
237      * @param mapTileWidth  the tile width specified in the map file
238      * @param mapTileHeight the tile height specified in the map file
239      */
240     public void render(int x, int y, int sx, int sy, int width, int ty,
241                        boolean lineByLine, int mapTileWidth, int mapTileHeight) {
242         for (int tileset = 0; tileset < map.getTileSetCount(); tileset++) {
243             TileSet set = null;
244 
245             for (int tx = 0; tx < width; tx++) {
246                 if ((sx + tx < 0) || (sy + ty < 0)) {
247                     continue;
248                 }
249                 if ((sx + tx >= this.width) || (sy + ty >= this.height)) {
250                     continue;
251                 }
252 
253                 if (data[sx + tx][sy + ty][0] == tileset) {
254                     if (set == null) {
255                         set = map.getTileSet(tileset);
256                         set.tiles.startUse();
257                     }
258 
259                     int sheetX = set.getTileX(data[sx + tx][sy + ty][1]);
260                     int sheetY = set.getTileY(data[sx + tx][sy + ty][1]);
261 
262                     int tileOffsetY = set.tileHeight - mapTileHeight;
263 
264                     // TODO CHECK
265                     // LSB: Rotate
266                     // LSB+1: Flip Y
267                     // LSB+2: Flip X
268                     byte b = (byte) ((data[sx + tx][sy + ty][2] & 0xE0000000L) >> 29);
269                     set.tiles.setAlpha(this.opacity); // Sets opacity/alpha value
270                     set.tiles.renderInUse(x + (tx * mapTileWidth), y
271                             + (ty * mapTileHeight) - tileOffsetY, sheetX,
272                             sheetY, b);
273                 }
274             }
275 
276             if (lineByLine) {
277                 if (set != null) {
278                     set.tiles.endUse();
279                     set = null;
280                 }
281                 map.renderedLine(ty, ty + sy, index);
282             }
283 
284             if (set != null) {
285                 set.tiles.endUse();
286             }
287         }
288     }

290     /**
291      * Decode a Base64 string as encoded by TilED
292      *
293      * @param data The string of character to decode
294      * @return The byte array represented by character encoding
295      */
296     private byte[] decodeBase64(char[] data) {
297         int temp = data.length;
298         for (int ix = 0; ix < data.length; ix++) {
299             if ((data[ix] > 255) || baseCodes[data[ix]] < 0) {
300                 --temp;
301             }
302         }
303 
304         int len = (temp / 4) * 3;
305         if ((temp % 4) == 3)
306             len += 2;
307         if ((temp % 4) == 2)
308             len += 1;
309 
310         byte[] out = new byte[len];
311 
312         int shift = 0;
313         int accum = 0;
314         int index = 0;
315 
316         for (int ix = 0; ix < data.length; ix++) {
317             int value = (data[ix] > 255) ? -1 : baseCodes[data[ix]];
318 
319             if (value >= 0) {
320                 accum <<= 6;
321                 shift += 6;
322                 accum |= value;
323                 if (shift >= 8) {
324                     shift -= 8;
325                     out[index++] = (byte) ((accum >> shift) & 0xff);
326                 }
327             }
328         }
329 
330         if (index != out.length) {
331             throw new RuntimeException(
332                     "Data length appears to be wrong (wrote " + index
333                             + " should be " + out.length + ")");
334         }
335 
336         return out;
337     }

339     /**
340      * Gets all Tiles from this layer, formatted into Tile objects Can only be
341      * used if the layer was loaded using TiledMapPlus
342      *
343      * @throws SlickException
344      * @author liamzebedee
345      */
346     public ArrayList<Tile> getTiles() throws SlickException {
347         if (tmap == null) {
348             throw new SlickException(
349                     "This method can only be used with Layers loaded using TiledMapPlus");
350         }
351         ArrayList<Tile> tiles = new ArrayList<Tile>();
352         for (int x = 0; x < this.width; x++) {
353             for (int y = 0; y < this.height; y++) {
354                 String tilesetName = tmap.tileSets.get(this.data[x][y][0]).name;
355                 Tile t = new Tile(x, y, this.name, y, tilesetName);
356                 tiles.add(t);
357             }
358         }
359         return tiles;
360     }

363      * Get all tiles from this layer that are part of a tileset Can only be used
364      * if the layer was loaded using TiledMapPlus

370     public ArrayList<Tile> getTilesOfTileset(String tilesetName)
371             throws SlickException {
372         if (tmap == null) {
373             throw new SlickException(
374                     "This method can only be used with Layers loaded using TiledMapPlus");
375         }
376         ArrayList<Tile> tiles = new ArrayList<Tile>();
377         int tilesetID = tmap.getTilesetID(tilesetName);
378         for (int x = 0; x < tmap.getWidth(); x++) {
379             for (int y = 0; y < tmap.getHeight(); y++) {
380                 if (this.data[x][y][0] == tilesetID) {
381                     Tile t = new Tile(x, y, this.name, this.data[x][y][1],
382                             tilesetName);
383                     tiles.add(t);
384                 }
385             }
386         }
387         return tiles;
