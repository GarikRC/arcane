8    * A heuristic that uses the tile that is closest to the target
9    * as the next best tile. In this case the sqrt is removed
10   * and the distance squared is used instead
11   *
12   * @author Kevin Glass
13   */
14  public class ClosestSquaredHeuristic implements AStarHeuristic {

16      /**
17       * @see AStarHeuristic#getCost(TileBasedMap, Mover, int, int, int, int)
18       */
19      public float getCost(TileBasedMap map, Mover mover, int x, int y, int tx, int ty) {
20          float dx = tx - x;
21          float dy = ty - y;
22  
23          return ((dx * dx) + (dy * dy));
24      }
