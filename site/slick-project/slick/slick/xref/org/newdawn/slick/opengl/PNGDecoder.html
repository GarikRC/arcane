101     public PNGDecoder(InputStream input) throws IOException {
102         this.input = input;
103         this.crc = new CRC32();
104         this.buffer = new byte[4096];
105 
106         readFully(buffer, 0, SIGNATURE.length);
107         if (!checkSignature(buffer)) {
108             throw new IOException("Not a valid PNG file");
109         }
110 
111         openChunk(IHDR);
112         readIHDR();
113         closeChunk();
114 
115         searchIDAT:
116         for (; ; ) {
117             openChunk();
118             switch (chunkType) {
119                 case IDAT:
120                     break searchIDAT;
121                 case PLTE:
122                     readPLTE();
123                     break;
124                 case tRNS:
125                     readtRNS();
126                     break;
127             }
128             closeChunk();
129         }
130 
131         if (colorType == COLOR_INDEXED && palette == null) {
132             throw new IOException("Missing PLTE chunk");
133         }
134     }

172     /**
173      * Overwrites the tRNS chunk entry to make a selected color transparent.
174      * <p>
175      * This can only be invoked when the image has no alpha channel.
176      * </p>
177      * <p>
178      * Calling this method causes {@link #hasAlpha()} to return true.
179      * </p>
180      *
181      * @param r the red component of the color to make transparent
182      * @param g the green component of the color to make transparent
183      * @param b the blue component of the color to make transparent
184      * @throws UnsupportedOperationException if the tRNS chunk data can't be set
185      * @see #hasAlphaChannel()
186      */
187     public void overwriteTRNS(byte r, byte g, byte b) {
188         if (hasAlphaChannel()) {
189             throw new UnsupportedOperationException(
190                     "image has an alpha channel");
191         }
192         byte[] pal = this.palette;
193         if (pal == null) {
194             transPixel = new byte[]{0, r, 0, g, 0, b};
195         } else {
196             paletteA = new byte[pal.length / 3];
197             for (int i = 0, j = 0; i < pal.length; i += 3, j++) {
198                 if (pal[i] != r || pal[i + 1] != g || pal[i + 2] != b) {
199                     paletteA[j] = (byte) 0xFF;
200                 }
201             }
202         }
203     }

205     /**
206      * Computes the implemented format conversion for the desired format.
207      *
208      * @param fmt the desired format
209      * @return format which best matches the desired format
210      * @throws UnsupportedOperationException if this PNG file can't be decoded
211      */
212     public Format decideTextureFormat(Format fmt) {
213         switch (colorType) {
214             case COLOR_TRUECOLOR:
215                 switch (fmt) {
216                     case ABGR:
217                     case RGBA:
218                     case BGRA:
219                     case RGB:
220                         return fmt;
221                     default:
222                         return Format.RGB;
223                 }
224             case COLOR_TRUEALPHA:
225                 switch (fmt) {
226                     case ABGR:
227                     case RGBA:
228                     case BGRA:
229                     case RGB:
230                         return fmt;
231                     default:
232                         return Format.RGBA;
233                 }
234             case COLOR_GREYSCALE:
235                 switch (fmt) {
236                     case LUMINANCE:
237                     case ALPHA:
238                         return fmt;
239                     default:
240                         return Format.LUMINANCE;
241                 }
242             case COLOR_GREYALPHA:
243                 return Format.LUMINANCE_ALPHA;
244             case COLOR_INDEXED:
245                 switch (fmt) {
246                     case ABGR:
247                     case RGBA:
248                     case BGRA:
249                         return fmt;
250                     default:
251                         return Format.RGBA;
252                 }
253             default:
254                 throw new UnsupportedOperationException("Not yet implemented");
255         }

258     /**
259      * Decodes the image into the specified buffer. The first line is placed at
260      * the current position. After decode the buffer position is at the end of
261      * the last line.
262      *
263      * @param buffer the buffer
264      * @param stride the stride in bytes from start of a line to start of the next
265      *               line, can be negative.
266      * @param fmt    the target format into which the image should be decoded.
267      * @throws IOException                   if a read or data error occurred
268      * @throws IllegalArgumentException      if the start position of a line falls outside the buffer
269      * @throws UnsupportedOperationException if the image can't be decoded into the desired format
270      */
271     public void decode(ByteBuffer buffer, int stride, Format fmt)
272             throws IOException {
273         final int offset = buffer.position();
274         final int lineSize = ((width * bitdepth + 7) / 8) * bytesPerPixel;
275         byte[] curLine = new byte[lineSize + 1];
276         byte[] prevLine = new byte[lineSize + 1];
277         byte[] palLine = (bitdepth < 8) ? new byte[width + 1] : null;
278 
279         final Inflater inflater = new Inflater();
280         try {
281             for (int y = 0; y < height; y++) {
282                 readChunkUnzip(inflater, curLine, 0, curLine.length);
283                 unfilter(curLine, prevLine);
284 
285                 buffer.position(offset + y * stride);
286 
287                 switch (colorType) {
288                     case COLOR_TRUECOLOR:
289                         switch (fmt) {
290                             case ABGR:
291                                 copyRGBtoABGR(buffer, curLine);
292                                 break;
293                             case RGBA:
294                                 copyRGBtoRGBA(buffer, curLine);
295                                 break;
296                             case BGRA:
297                                 copyRGBtoBGRA(buffer, curLine);
298                                 break;
299                             case RGB:
300                                 copy(buffer, curLine);
301                                 break;
302                             default:
303                                 throw new UnsupportedOperationException(
304                                         "Unsupported format for this image");
305                         }
306                         break;
307                     case COLOR_TRUEALPHA:
308                         switch (fmt) {
309                             case ABGR:
310                                 copyRGBAtoABGR(buffer, curLine);
311                                 break;
312                             case RGBA:
313                                 copy(buffer, curLine);
314                                 break;
315                             case BGRA:
316                                 copyRGBAtoBGRA(buffer, curLine);
317                                 break;
318                             case RGB:
319                                 copyRGBAtoRGB(buffer, curLine);
320                                 break;
321                             default:
322                                 throw new UnsupportedOperationException(
323                                         "Unsupported format for this image");
324                         }
325                         break;
326                     case COLOR_GREYSCALE:
327                         switch (fmt) {
328                             case LUMINANCE:
329                             case ALPHA:
330                                 copy(buffer, curLine);
331                                 break;
332                             default:
333                                 throw new UnsupportedOperationException(
334                                         "Unsupported format for this image");
335                         }
336                         break;
337                     case COLOR_GREYALPHA:
338                         switch (fmt) {
339                             case LUMINANCE_ALPHA:
340                                 copy(buffer, curLine);
341                                 break;
342                             default:
343                                 throw new UnsupportedOperationException(
344                                         "Unsupported format for this image");
345                         }
346                         break;
347                     case COLOR_INDEXED:
348                         switch (bitdepth) {
349                             case 8:
350                                 palLine = curLine;
351                                 break;
352                             case 4:
353                                 expand4(curLine, palLine);
354                                 break;
355                             case 2:
356                                 expand2(curLine, palLine);
357                                 break;
358                             case 1:
359                                 expand1(curLine, palLine);
360                                 break;
361                             default:
362                                 throw new UnsupportedOperationException(
363                                         "Unsupported bitdepth for this image");
364                         }
365                         switch (fmt) {
366                             case ABGR:
367                                 copyPALtoABGR(buffer, palLine);
368                                 break;
369                             case RGBA:
370                                 copyPALtoRGBA(buffer, palLine);
371                                 break;
372                             case BGRA:
373                                 copyPALtoBGRA(buffer, palLine);
374                                 break;
375                             default:
376                                 throw new UnsupportedOperationException(
377                                         "Unsupported format for this image");
378                         }
379                         break;
380                     default:
381                         throw new UnsupportedOperationException(
382                                 "Not yet implemented");
383                 }
384 
385                 byte[] tmp = curLine;
386                 curLine = prevLine;
387                 prevLine = tmp;
388             }
389         } finally {
390             inflater.end();
391         }
392     }

394     /**
395      * Decodes the image into the specified buffer. The last line is placed at
396      * the current position. After decode the buffer position is at the end of
397      * the first line.
398      *
399      * @param buffer the buffer
400      * @param stride the stride in bytes from start of a line to start of the next
401      *               line, must be positive.
402      * @param fmt    the target format into which the image should be decoded.
403      * @throws IOException                   if a read or data error occurred
404      * @throws IllegalArgumentException      if the start position of a line falls outside the buffer
405      * @throws UnsupportedOperationException if the image can't be decoded into the desired format
406      */
407     public void decodeFlipped(ByteBuffer buffer, int stride, Format fmt)
408             throws IOException {
409         if (stride <= 0) {
410             throw new IllegalArgumentException("stride");
411         }
412         int pos = buffer.position();
413         int posDelta = (height - 1) * stride;
414         buffer.position(pos + posDelta);
415         decode(buffer, -stride, fmt);
416         buffer.position(buffer.position() + posDelta);
417     }

423     private void copyRGBtoABGR(ByteBuffer buffer, byte[] curLine) {
424         if (transPixel != null) {
425             byte tr = transPixel[1];
426             byte tg = transPixel[3];
427             byte tb = transPixel[5];
428             for (int i = 1, n = curLine.length; i < n; i += 3) {
429                 byte r = curLine[i];
430                 byte g = curLine[i + 1];
431                 byte b = curLine[i + 2];
432                 byte a = (byte) 0xFF;
433                 if (r == tr && g == tg && b == tb) {
434                     a = 0;
435                 }
436                 buffer.put(a).put(b).put(g).put(r);
437             }
438         } else {
439             for (int i = 1, n = curLine.length; i < n; i += 3) {
440                 buffer.put((byte) 0xFF).put(curLine[i + 2]).put(curLine[i + 1])
441                         .put(curLine[i]);
442             }
443         }
444     }

578     private void expand4(byte[] src, byte[] dst) {
579         for (int i = 1, n = dst.length; i < n; i += 2) {
580             int val = src[1 + (i >> 1)] & 255;
581             switch (n - i) {
582                 default:
583                     dst[i + 1] = (byte) (val & 15);
584                 case 1:
585                     dst[i] = (byte) (val >> 4);
586             }
587         }
588     }
589 
590     private void expand2(byte[] src, byte[] dst) {
591         for (int i = 1, n = dst.length; i < n; i += 4) {
592             int val = src[1 + (i >> 2)] & 255;
593             switch (n - i) {
594                 default:
595                     dst[i + 3] = (byte) ((val) & 3);
596                 case 3:
597                     dst[i + 2] = (byte) ((val >> 2) & 3);
598                 case 2:
599                     dst[i + 1] = (byte) ((val >> 4) & 3);
600                 case 1:
601                     dst[i] = (byte) ((val >> 6));
602             }
603         }
604     }
605 
606     private void expand1(byte[] src, byte[] dst) {
607         for (int i = 1, n = dst.length; i < n; i += 8) {
608             int val = src[1 + (i >> 3)] & 255;
609             switch (n - i) {
610                 default:
611                     dst[i + 7] = (byte) ((val) & 1);
612                 case 7:
613                     dst[i + 6] = (byte) ((val >> 1) & 1);
614                 case 6:
615                     dst[i + 5] = (byte) ((val >> 2) & 1);
616                 case 5:
617                     dst[i + 4] = (byte) ((val >> 3) & 1);
618                 case 4:
619                     dst[i + 3] = (byte) ((val >> 4) & 1);
620                 case 3:
621                     dst[i + 2] = (byte) ((val >> 5) & 1);
622                 case 2:
623                     dst[i + 1] = (byte) ((val >> 6) & 1);
624                 case 1:
625                     dst[i] = (byte) ((val >> 7));
626             }
627         }
628     }

630     private void unfilter(byte[] curLine, byte[] prevLine) throws IOException {
631         switch (curLine[0]) {
632             case 0: // none
633                 break;
634             case 1:
635                 unfilterSub(curLine);
636                 break;
637             case 2:
638                 unfilterUp(curLine, prevLine);
639                 break;
640             case 3:
641                 unfilterAverage(curLine, prevLine);
642                 break;
643             case 4:
644                 unfilterPaeth(curLine, prevLine);
645                 break;
646             default:
647                 throw new IOException("invalide filter type in scanline: "
648                         + curLine[0]);
649         }
650     }
651 
652     private void unfilterSub(byte[] curLine) {
653         final int bpp = this.bytesPerPixel;
654         for (int i = bpp + 1, n = curLine.length; i < n; ++i) {
655             curLine[i] += curLine[i - bpp];
656         }
657     }
658 
659     private void unfilterUp(byte[] curLine, byte[] prevLine) {
660         final int bpp = this.bytesPerPixel;
661         for (int i = 1, n = curLine.length; i < n; ++i) {
662             curLine[i] += prevLine[i];
663         }
664     }
665 
666     private void unfilterAverage(byte[] curLine, byte[] prevLine) {
667         final int bpp = this.bytesPerPixel;
668 
669         int i;
670         for (i = 1; i <= bpp; ++i) {
671             curLine[i] += (byte) ((prevLine[i] & 0xFF) >>> 1);
672         }
673         for (int n = curLine.length; i < n; ++i) {
674             curLine[i] += (byte) (((prevLine[i] & 0xFF) + (curLine[i - bpp] & 0xFF)) >>> 1);
675         }
676     }

678     private void unfilterPaeth(byte[] curLine, byte[] prevLine) {
679         final int bpp = this.bytesPerPixel;
680 
681         int i;
682         for (i = 1; i <= bpp; ++i) {
683             curLine[i] += prevLine[i];
684         }
685         for (int n = curLine.length; i < n; ++i) {
686             int a = curLine[i - bpp] & 255;
687             int b = prevLine[i] & 255;
688             int c = prevLine[i - bpp] & 255;
689             int p = a + b - c;
690             int pa = p - a;
691             if (pa < 0)
692                 pa = -pa;
693             int pb = p - b;
694             if (pb < 0)
695                 pb = -pb;
696             int pc = p - c;
697             if (pc < 0)
698                 pc = -pc;
699             if (pa <= pb && pa <= pc)
700                 c = a;

803         if (chunkRemaining > 0) {

843             throws IOException {
844         if (length > chunkRemaining) {
845             length = chunkRemaining;
