7   /**
8    * A utility to allow performing GL operations without contaminating the
9    * Slick OpenGL state. Note this will not protect you from OpenGL programming errors
10   * like a glBegin() without a glEnd(), or glPush() without glPop() etc.
11   * <p/>
12   * Expected usage:
13   * <p/>
14   * <code>
15   * SlickCallable callable = new SlickCallable() {
16   * public performGLOperations() throws SlickException {
17   * GL.glTranslate(0,0,1);
18   * glBegin(GL.GL_POLYGONS);
19   * glVertex(..);
20   * ...
21   * glEnd();
22   * }
23   * }
24   * callable.call();
25   * </code>
26   * <p/>
27   * Alternatively you can use the static methods directly
28   * <p/>
29   * <code>
30   * SlickCallable.enterSafeBlock();
31   * <p/>
32   * GL.glTranslate(0,0,1);
33   * glBegin(GL.GL_POLYGONS);
34   * glVertex(..);
35   * ...
36   * glEnd();
37   * <p/>
38   * SlickCallable.leaveSafeBlock();
39   * </code>
40   *
41   * @author kevin
42   */
43  public abstract class SlickCallable {
44      /**
45       * The last texture used
46       */
47      private static Texture lastUsed;
48      /**
49       * True if we're in a safe block
50       */
51      private static boolean inSafe = false;
52  
53  
54      /**
55       * Enter a safe block ensuring that all the OpenGL state that slick
56       * uses is safe before touching the GL state directly.
57       */
58      public static void enterSafeBlock() {
59          if (inSafe) {
60              return;
61          }
62  
63          Renderer.get().flush();
64          lastUsed = TextureImpl.getLastBind();
65          TextureImpl.bindNone();
66          GL11.glPushAttrib(GL11.GL_ALL_ATTRIB_BITS);
67          GL11.glPushClientAttrib(GL11.GL_ALL_CLIENT_ATTRIB_BITS);
68          GL11.glMatrixMode(GL11.GL_MODELVIEW);
69          GL11.glPushMatrix();
70          GL11.glMatrixMode(GL11.GL_PROJECTION);
71          GL11.glPushMatrix();
72          GL11.glMatrixMode(GL11.GL_MODELVIEW);
73  
74          inSafe = true;
75      }
76  
77      /**
78       * Leave a safe block ensuring that all of Slick's OpenGL state is
79       * restored since the last enter.
80       */
81      public static void leaveSafeBlock() {
82          if (!inSafe) {
83              return;
84          }
85  
86          GL11.glMatrixMode(GL11.GL_PROJECTION);
87          GL11.glPopMatrix();
88          GL11.glMatrixMode(GL11.GL_MODELVIEW);
89          GL11.glPopMatrix();
90          GL11.glPopClientAttrib();
91          GL11.glPopAttrib();
92  
93          if (lastUsed != null) {
94              lastUsed.bind();
95          } else {
96              TextureImpl.bindNone();
97          }
98  
99          inSafe = false;
100     }
101 
102     /**
103      * Cause this callable to perform it's GL operations (@see performGLOperations()). This
104      * method will block until the GL operations have been performed.
105      *
106      * @throws SlickException Indicates a failure while performing the GL operations or
107      *                        maintaing SlickState
108      */
109     public final void call() throws SlickException {
110         enterSafeBlock();
111 
112         performGLOperations();
113 
114         leaveSafeBlock();
115     }
116 
117     /**
118      * Perform the GL operations that this callable is intended to. This operations should
119      * not effect the slick OpenGL state.
120      *
121      * @throws SlickException Indicates a failure of some sort. This is user exception
122      */
123     protected abstract void performGLOperations() throws SlickException;
124 }
