228     /**
229      * Set the drawing mode to use. This mode defines how pixels are drawn to the graphics context. It can be used to
230      * draw into the alpha map.
231      * <p/>
232      * The mode supplied should be one of {@link Graphics#MODE_NORMAL} or {@link Graphics#MODE_ALPHA_MAP} or
233      * {@link Graphics#MODE_ALPHA_BLEND}
234      *
235      * @param mode The mode to apply.
236      */
237     public void setDrawMode(int mode) {
238         predraw();
239         currentDrawingMode = mode;
240         if (currentDrawingMode == MODE_NORMAL) {
241             GL.glEnable(SGL.GL_BLEND);
242             GL.glColorMask(true, true, true, true);
243             GL.glBlendFunc(SGL.GL_SRC_ALPHA, SGL.GL_ONE_MINUS_SRC_ALPHA);
244         }
245         if (currentDrawingMode == MODE_ALPHA_MAP) {
246             GL.glDisable(SGL.GL_BLEND);
247             GL.glColorMask(false, false, false, true);
248         }
249         if (currentDrawingMode == MODE_ALPHA_BLEND) {
250             GL.glEnable(SGL.GL_BLEND);
251             GL.glColorMask(true, true, true, false);
252             GL.glBlendFunc(SGL.GL_DST_ALPHA, SGL.GL_ONE_MINUS_DST_ALPHA);
253         }
254         if (currentDrawingMode == MODE_COLOR_MULTIPLY) {
255             GL.glEnable(SGL.GL_BLEND);
256             GL.glColorMask(true, true, true, true);
257             GL.glBlendFunc(SGL.GL_ONE_MINUS_SRC_COLOR, SGL.GL_SRC_COLOR);
258         }
259         if (currentDrawingMode == MODE_ADD) {
260             GL.glEnable(SGL.GL_BLEND);
261             GL.glColorMask(true, true, true, true);
262             GL.glBlendFunc(SGL.GL_ONE, SGL.GL_ONE);
263         }
264         if (currentDrawingMode == MODE_SCREEN) {
265             GL.glEnable(SGL.GL_BLEND);
266             GL.glColorMask(true, true, true, true);
267             GL.glBlendFunc(SGL.GL_ONE, SGL.GL_ONE_MINUS_SRC_COLOR);
268         }
269         if (currentDrawingMode == MODE_ADD_ALPHA) {
270             GL.glEnable(SGL.GL_BLEND);
271             GL.glColorMask(true, true, true, true);
272             GL.glBlendFunc(SGL.GL_SRC_ALPHA, SGL.GL_ONE);
273         }
274         if (currentDrawingMode == MODE_COLOR_MULTIPLY_ALPHA) {
275             GL.glEnable(SGL.GL_BLEND);
276             GL.glColorMask(true, true, true, true);
277             GL.glBlendFunc(SGL.GL_ONE_MINUS_SRC_COLOR, SGL.GL_ONE_MINUS_SRC_ALPHA);
278         }
279         postdraw();
280     }

495     /**
496      * Draw a line on the canvas in the current colour
497      *
498      * @param x1 The x coordinate of the start point
499      * @param y1 The y coordinate of the start point
500      * @param x2 The x coordinate of the end point
501      * @param y2 The y coordinate of the end point
502      */
503     public void drawLine(float x1, float y1, float x2, float y2) {
504         float lineWidth = this.lineWidth - 1;
505 
506         if (LSR.applyGLLineFixes()) {
507             if (x1 == x2) {
508                 if (y1 > y2) {
509                     float temp = y2;
510                     y2 = y1;
511                     y1 = temp;
512                 }
513                 float step = 1 / sy;
514                 lineWidth = lineWidth / sy;
515                 fillRect(x1 - (lineWidth / 2.0f), y1 - (lineWidth / 2.0f), lineWidth + step, (y2 - y1) + lineWidth + step);
516                 return;
517             } else if (y1 == y2) {
518                 if (x1 > x2) {
519                     float temp = x2;
520                     x2 = x1;
521                     x1 = temp;
522                 }
523                 float step = 1 / sx;
524                 lineWidth = lineWidth / sx;
525                 fillRect(x1 - (lineWidth / 2.0f), y1 - (lineWidth / 2.0f), (x2 - x1) + lineWidth + step, lineWidth + step);
526                 return;
527             }
528         }
529 
530         predraw();
531         currentColor.bind();
532         TextureImpl.bindNone();
533 
534         LSR.start();
535         LSR.vertex(x1, y1);
536         LSR.vertex(x2, y2);
537         LSR.end();
538 
539         postdraw();
540     }

721     /**
722      * Set clipping that controls which areas of the world will be drawn to. Note that world clip is different from
723      * standard screen clip in that it's defined in the space of the current world coordinate - i.e. it's affected by
724      * translate, rotate, scale etc.
725      *
726      * @param x      The x coordinate of the top left corner of the allowed area
727      * @param y      The y coordinate of the top left corner of the allowed area
728      * @param width  The width of the allowed area
729      * @param height The height of the allowed area
730      */
731     public void setWorldClip(float x, float y, float width, float height) {
732         predraw();
733         worldClipRecord = new Rectangle(x, y, width, height);
734 
735         GL.glEnable(SGL.GL_CLIP_PLANE0);
736         worldClip.put(1).put(0).put(0).put(-x).flip();
737         GL.glClipPlane(SGL.GL_CLIP_PLANE0, worldClip);
738         GL.glEnable(SGL.GL_CLIP_PLANE1);
739         worldClip.put(-1).put(0).put(0).put(x + width).flip();
740         GL.glClipPlane(SGL.GL_CLIP_PLANE1, worldClip);
741 
742         GL.glEnable(SGL.GL_CLIP_PLANE2);
743         worldClip.put(0).put(1).put(0).put(-y).flip();
744         GL.glClipPlane(SGL.GL_CLIP_PLANE2, worldClip);
745         GL.glEnable(SGL.GL_CLIP_PLANE3);
746         worldClip.put(0).put(-1).put(0).put(y + height).flip();
747         GL.glClipPlane(SGL.GL_CLIP_PLANE3, worldClip);
748         postdraw();
749     }

1008      * @param y1       The y coordinate of the top left corner of a box containing the arc
1009      * @param width    The width of the arc

1012      * @param start    The angle the arc starts at

1018         currentColor.bind();
1019

1022         }
1023

1025         float cy = y1 + (height / 2.0f);
1026

1028         int step = 360 / segments;
1029 
1030         GL.glVertex2f(cx, cy);
