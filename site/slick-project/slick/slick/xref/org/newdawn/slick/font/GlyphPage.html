138     /**
139      * Loads glyphs to the backing texture and sets the image on each loaded glyph. Loaded glyphs are removed from the list.
140      * <p/>
141      * If this page already has glyphs and maxGlyphsToLoad is -1, then this method will return 0 if all the new glyphs don't fit.
142      * This reduces texture binds when drawing since glyphs loaded at once are typically displayed together.
143      *
144      * @param glyphs          The glyphs to load.
145      * @param maxGlyphsToLoad This is the maximum number of glyphs to load from the list. Set to -1 to attempt to load all the
146      *                        glyphs.
147      * @return The number of glyphs that were actually loaded.
148      * @throws SlickException if the glyph could not be rendered.
149      */
150     public int loadGlyphs(List glyphs, int maxGlyphsToLoad) throws SlickException {
151         if (rowHeight != 0 && maxGlyphsToLoad == -1) {
152             // If this page has glyphs and we are not loading incrementally, return zero if any of the glyphs don't fit.
153             int testX = pageX;
154             int testY = pageY;
155             int testRowHeight = rowHeight;
156             for (Iterator iter = getIterator(glyphs); iter.hasNext(); ) {
157                 Glyph glyph = (Glyph) iter.next();
158                 int width = glyph.getWidth();
159                 int height = glyph.getHeight();
160                 if (testX + width >= pageWidth) {
161                     testX = 0;
162                     testY += testRowHeight;
163                     testRowHeight = height;
164                 } else if (height > testRowHeight) {
165                     testRowHeight = height;
166                 }
167                 if (testY + testRowHeight >= pageWidth) return 0;
168                 testX += width;
169             }
170         }
171 
172         Color.white.bind();
173         pageImage.bind();
174 
175         int i = 0;
176         for (Iterator iter = getIterator(glyphs); iter.hasNext(); ) {
177             Glyph glyph = (Glyph) iter.next();
178             int width = Math.min(MAX_GLYPH_SIZE, glyph.getWidth());
179             int height = Math.min(MAX_GLYPH_SIZE, glyph.getHeight());
180 
181             if (rowHeight == 0) {
182                 // The first glyph always fits.
183                 rowHeight = height;
184             } else {
185                 // Wrap to the next line if needed, or break if no more fit.
186                 if (pageX + width >= pageWidth) {
187                     if (pageY + rowHeight + height >= pageHeight) break;
188                     pageX = 0;
189                     pageY += rowHeight;
190                     rowHeight = height;
191                 } else if (height > rowHeight) {
192                     if (pageY + height >= pageHeight) break;
193                     rowHeight = height;
194                 }
195             }
196 
197             renderGlyph(glyph, width, height);
198             pageGlyphs.add(glyph);
199 
200             pageX += width;
201 
202             iter.remove();
203             i++;
204             if (i == maxGlyphsToLoad) {
205                 // If loading incrementally, flip orderAscending so it won't change, since we'll probably load the rest next time.
206                 orderAscending = !orderAscending;
207                 break;
208             }
209         }
210 
211         TextureImpl.bindNone();
212 
213         // Every other batch of glyphs added to a page are sorted the opposite way to attempt to keep same size glyps together.
214         orderAscending = !orderAscending;
215 
216         return i;
217     }

219     /**
220      * Loads a single glyph to the backing texture, if it fits.
221      *
222      * @param glyph  The glyph to be rendered
223      * @param width  The expected width of the glyph
224      * @param height The expected height of the glyph
225      * @throws SlickException if the glyph could not be rendered.
226      */
227     private void renderGlyph(Glyph glyph, int width, int height) throws SlickException {
228         // Draw the glyph to the scratch image using Java2D.
229         scratchGraphics.setComposite(AlphaComposite.Clear);
230         scratchGraphics.fillRect(0, 0, MAX_GLYPH_SIZE, MAX_GLYPH_SIZE);
231         scratchGraphics.setComposite(AlphaComposite.SrcOver);
232         scratchGraphics.setColor(java.awt.Color.white);
233         for (Iterator iter = unicodeFont.getEffects().iterator(); iter.hasNext(); )
234             ((Effect) iter.next()).draw(scratchImage, scratchGraphics, unicodeFont, glyph);
235         glyph.setShape(null); // The shape will never be needed again.
236 
237         WritableRaster raster = scratchImage.getRaster();
238         int[] row = new int[width];
239         for (int y = 0; y < height; y++) {
240             raster.getDataElements(0, y, width, 1, row);
241             scratchIntBuffer.put(row);
242         }
243         GL.glTexSubImage2D(SGL.GL_TEXTURE_2D, 0, pageX, pageY, width, height, SGL.GL_BGRA, SGL.GL_UNSIGNED_BYTE,
244                 scratchByteBuffer);
245         scratchIntBuffer.clear();
246 
247         glyph.setImage(pageImage.getSubImage(pageX, pageY, width, height));
248     }

250     /**
251      * Returns an iterator for the specified glyphs, sorted either ascending or descending.
252      *
253      * @param glyphs The glyphs to return if present
254      * @return An iterator of the sorted list of glyphs
255      */
256     private Iterator getIterator(List glyphs) {
257         if (orderAscending) return glyphs.iterator();
258         final ListIterator iter = glyphs.listIterator(glyphs.size());
259         return new Iterator() {
260             public boolean hasNext() {
261                 return iter.hasPrevious();
262             }
263 
264             public Object next() {
265                 return iter.previous();
266             }
267 
268             public void remove() {
269                 iter.remove();
270             }
271         };
272     }
