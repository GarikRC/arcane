27  /**
28   * A Slick bitmap font that can display unicode glyphs from a TrueTypeFont.
29   * <p/>
30   * For efficiency, glyphs are packed on to textures. Glyphs can be loaded to the textures on the fly, when they are first needed
31   * for display. However, it is best to load the glyphs that are known to be needed at startup.
32   *
33   * @author Nathan Sweet <misc@n4te.com>
34   */

63       *
64       * @param ttfFileRef The file system or classpath location of the TrueTypeFont file.
65       * @return The font created
66       * @throws SlickException Indicates a failure to locate or load the font into Java's font

69      private static Font createFont(String ttfFileRef) throws SlickException {
70          try {
71              return Font.createFont(Font.TRUETYPE_FONT, ResourceLoader.getResourceAsStream(ttfFileRef));
72          } catch (FontFormatException ex) {
73              throw new SlickException("Invalid font: " + ttfFileRef, ex);
74          } catch (IOException ex) {
75              throw new SlickException("Error reading font: " + ttfFileRef, ex);
76          }
77      }

278     /**
279      * Initialise the font to be used based on configuration
280      *
281      * @param baseFont The AWT font to render
282      * @param size     The point size of the font to generated
283      * @param bold     True if the font should be rendered in bold typeface
284      * @param italic   True if the font should be rendered in bold typeface
285      */
286     private void initializeFont(Font baseFont, int size, boolean bold, boolean italic) {
287         Map attributes = baseFont.getAttributes();
288         attributes.put(TextAttribute.SIZE, new Float(size));
289         attributes.put(TextAttribute.WEIGHT, bold ? TextAttribute.WEIGHT_BOLD : TextAttribute.WEIGHT_REGULAR);
290         attributes.put(TextAttribute.POSTURE, italic ? TextAttribute.POSTURE_OBLIQUE : TextAttribute.POSTURE_REGULAR);
291         try {
292             attributes.put(TextAttribute.class.getDeclaredField("KERNING").get(null), TextAttribute.class.getDeclaredField(
293                     "KERNING_ON").get(null));
294         } catch (Exception ignored) {
295         }
296         font = baseFont.deriveFont(attributes);
297 
298         FontMetrics metrics = GlyphPage.getScratchGraphics().getFontMetrics(font);
299 
300         ascent = metrics.getAscent();
301         descent = metrics.getDescent();
302         leading = metrics.getLeading();
303 
304         // Determine width of space glyph (getGlyphPixelBounds gives a width of zero).
305         char[] chars = " ".toCharArray();
306         GlyphVector vector = font.layoutGlyphVector(GlyphPage.renderContext, chars, 0, chars.length, Font.LAYOUT_LEFT_TO_RIGHT);
307         spaceWidth = vector.getGlyphLogicalBounds(0).getBounds().width;
308     }

327     /**
328      * Queues the glyphs in the specified codepoint range (inclusive) to be loaded. Note that the glyphs are not actually loaded
329      * until {@link #loadGlyphs()} is called.
330      * <p/>
331      * Some characters like combining marks and non-spacing marks can only be rendered with the context of other glyphs. In this
332      * case, use {@link #addGlyphs(String)}.
333      *
334      * @param startCodePoint The code point of the first glyph to add
335      * @param endCodePoint   The code point of the last glyph to add
336      */
337     public void addGlyphs(int startCodePoint, int endCodePoint) {
338         for (int codePoint = startCodePoint; codePoint <= endCodePoint; codePoint++)
339             addGlyphs(new String(Character.toChars(codePoint)));
340     }

342     /**
343      * Queues the glyphs in the specified text to be loaded. Note that the glyphs are not actually loaded until
344      * {@link #loadGlyphs()} is called.
345      *
346      * @param text The text containing the glyphs to be added
347      */
348     public void addGlyphs(String text) {
349         if (text == null) throw new IllegalArgumentException("text cannot be null.");
350 
351         char[] chars = text.toCharArray();
352         GlyphVector vector = font.layoutGlyphVector(GlyphPage.renderContext, chars, 0, chars.length, Font.LAYOUT_LEFT_TO_RIGHT);
353         for (int i = 0, n = vector.getNumGlyphs(); i < n; i++) {
354             int codePoint = text.codePointAt(vector.getGlyphCharIndex(i));
355             Rectangle bounds = getGlyphBounds(vector, i, codePoint);
356             getGlyph(vector.getGlyphCode(i), codePoint, bounds, vector, i);
357         }
358     }

376     /**
377      * Loads all queued glyphs to the backing textures. Glyphs that are typically displayed together should be added and loaded at
378      * the same time so that they are stored on the same backing texture. This reduces the number of backing texture binds required
379      * to draw glyphs.
380      *
381      * @return True if the glyphs were loaded entirely
382      * @throws SlickException if the glyphs could not be loaded.
383      */
384     public boolean loadGlyphs() throws SlickException {
385         return loadGlyphs(-1);
386     }

388     /**
389      * Loads up to the specified number of queued glyphs to the backing textures. This is typically called from the game loop to
390      * load glyphs on the fly that were requested for display but have not yet been loaded.
391      *
392      * @param maxGlyphsToLoad The maximum number of glyphs to be loaded this time
393      * @return True if the glyphs were loaded entirely
394      * @throws SlickException if the glyphs could not be loaded.
395      */
396     public boolean loadGlyphs(int maxGlyphsToLoad) throws SlickException {
397         if (queuedGlyphs.isEmpty()) return false;
398 
399         if (effects.isEmpty())
400             throw new IllegalStateException("The UnicodeFont must have at least one effect before any glyphs can be loaded.");
401 
402         for (Iterator iter = queuedGlyphs.iterator(); iter.hasNext(); ) {
403             Glyph glyph = (Glyph) iter.next();
404             int codePoint = glyph.getCodePoint();
405 
406             // Don't load an image for a glyph with nothing to display.
407             if (glyph.getWidth() == 0 || codePoint == ' ') {
408                 iter.remove();
409                 continue;
410             }
411 
412             // Only load the first missing glyph.
413             if (glyph.isMissing()) {
414                 if (missingGlyph != null) {
415                     if (glyph != missingGlyph) iter.remove();
416                     continue;
417                 }
418                 missingGlyph = glyph;
419             }
420         }
421 
422         Collections.sort(queuedGlyphs, heightComparator);
423 
424         // Add to existing pages.
425         for (Iterator iter = glyphPages.iterator(); iter.hasNext(); ) {
426             GlyphPage glyphPage = (GlyphPage) iter.next();
427             maxGlyphsToLoad -= glyphPage.loadGlyphs(queuedGlyphs, maxGlyphsToLoad);
428             if (maxGlyphsToLoad == 0 || queuedGlyphs.isEmpty())
429                 return true;
430         }
431 
432         // Add to new pages.
433         while (!queuedGlyphs.isEmpty()) {
434             GlyphPage glyphPage = new GlyphPage(this, glyphPageWidth, glyphPageHeight);
435             glyphPages.add(glyphPage);
436             maxGlyphsToLoad -= glyphPage.loadGlyphs(queuedGlyphs, maxGlyphsToLoad);
437             if (maxGlyphsToLoad == 0) return true;
438         }
439 
440         return true;
441     }

443     /**
444      * Clears all loaded and queued glyphs.
445      */
446     public void clearGlyphs() {
447         for (int i = 0; i < PAGES; i++)
448             glyphs[i] = null;
449 
450         for (Iterator iter = glyphPages.iterator(); iter.hasNext(); ) {
451             GlyphPage page = (GlyphPage) iter.next();
452             try {
453                 page.getImage().destroy();
454             } catch (SlickException ignored) {
455             }
456         }
457         glyphPages.clear();
458 
459         if (baseDisplayListID != -1) {
460             GL.glDeleteLists(baseDisplayListID, displayLists.size());
461             baseDisplayListID = -1;
462         }
463 
464         queuedGlyphs.clear();
465         missingGlyph = null;
466     }

477     /**
478      * Identical to {@link #drawString(float, float, String, Color, int, int)} but returns a
479      * DisplayList which provides access to the width and height of the text drawn.
480      *
481      * @param text       The text to render
482      * @param x          The horizontal location to render at
483      * @param y          The vertical location to render at
484      * @param color      The colour to apply as a filter on the text
485      * @param startIndex The start index into the string to start rendering at
486      * @param endIndex   The end index into the string to render to
487      * @return The reference to the display list that was drawn and potentiall ygenerated
488      */
489     public DisplayList drawDisplayList(float x, float y, CharSequence text, Color color, int startIndex, int endIndex) {
490         if (text == null) throw new IllegalArgumentException("text cannot be null.");
491         if (text.length() == 0) return EMPTY_DISPLAY_LIST;
492         if (color == null) throw new IllegalArgumentException("color cannot be null.");
493 
494         x -= paddingLeft;
495         y -= paddingTop;
496 
497         CharSequence displayListKey = text.subSequence(startIndex, endIndex);
498 
499         color.bind();
500         TextureImpl.bindNone();
501 
502         DisplayList displayList = null;
503         if (displayListCaching && queuedGlyphs.isEmpty()) {
504             if (baseDisplayListID == -1) {
505                 baseDisplayListID = GL.glGenLists(DISPLAY_LIST_CACHE_SIZE);
506                 if (baseDisplayListID == 0) {
507                     baseDisplayListID = -1;
508                     displayListCaching = false;
509                     return new DisplayList();
510                 }
511             }
512             // Try to use a display list compiled for this text.
513             displayList = (DisplayList) displayLists.get(displayListKey);
514             if (displayList != null) {
515                 if (displayList.invalid)
516                     displayList.invalid = false;
517                 else {
518                     GL.glTranslatef(x, y, 0);
519                     GL.glCallList(displayList.id);
520                     GL.glTranslatef(-x, -y, 0);
521                     return displayList;
522                 }
523             } else if (displayList == null) {
524                 // Compile a new display list.
525                 displayList = new DisplayList();
526                 int displayListCount = displayLists.size();
527                 displayLists.put(displayListKey, displayList);
528                 if (displayListCount < DISPLAY_LIST_CACHE_SIZE)
529                     displayList.id = baseDisplayListID + displayListCount;
530                 else
531                     displayList.id = eldestDisplayListID;
532             }
533             displayLists.put(displayListKey, displayList);
534         }
535 
536         GL.glTranslatef(x, y, 0);
537 
538         if (displayList != null) GL.glNewList(displayList.id, SGL.GL_COMPILE_AND_EXECUTE);
539 
540         char[] chars = toCharArray(text, 0, endIndex);
541         GlyphVector vector = font.layoutGlyphVector(GlyphPage.renderContext, chars, 0, chars.length, Font.LAYOUT_LEFT_TO_RIGHT);
542 
543         int maxWidth = 0, totalHeight = 0, lines = 0;
544         int extraX = 0, extraY = ascent;
545         boolean startNewLine = false;
546         Texture lastBind = null;
547         for (int glyphIndex = 0, n = vector.getNumGlyphs(); glyphIndex < n; glyphIndex++) {
548             int charIndex = vector.getGlyphCharIndex(glyphIndex);
549             if (charIndex < startIndex) continue;
550             if (charIndex > endIndex) break;
551 
552             int codePoint = Character.codePointAt(text, charIndex);
553 
554             Rectangle bounds = getGlyphBounds(vector, glyphIndex, codePoint);
555             Glyph glyph = getGlyph(vector.getGlyphCode(glyphIndex), codePoint, bounds, vector, glyphIndex);
556 
557             if (startNewLine && codePoint != '\n') {
558                 extraX = -bounds.x;
559                 startNewLine = false;
560             }
561 
562             Image image = glyph.getImage();
563             if (image == null && missingGlyph != null && glyph.isMissing()) image = missingGlyph.getImage();
564             if (image != null) {
565                 // Draw glyph, only binding a new glyph page texture when necessary.
566                 Texture texture = image.getTexture();
567                 if (lastBind != null && lastBind != texture) {
568                     GL.glEnd();
569                     lastBind = null;
570                 }
571                 if (lastBind == null) {
572                     texture.bind();
573                     GL.glBegin(SGL.GL_QUADS);
574                     lastBind = texture;
575                 }
576                 image.drawEmbedded(bounds.x + extraX, bounds.y + extraY, image.getWidth(), image.getHeight());
577             }
578 
579             if (glyphIndex >= 0) extraX += paddingRight + paddingLeft + paddingAdvanceX;
580             maxWidth = Math.max(maxWidth, bounds.x + extraX + bounds.width);
581             totalHeight = Math.max(totalHeight, ascent + bounds.y + bounds.height);
582 
583             if (codePoint == '\n') {
584                 startNewLine = true; // Mac gives -1 for bounds.x of '\n', so use the bounds.x of the next glyph.
585                 extraY += getLineHeight();
586                 lines++;
587                 totalHeight = 0;
588             }
589         }
590         if (lastBind != null) GL.glEnd();
591 
592         if (displayList != null) {
593             GL.glEndList();
594             // Invalidate the display list if it had glyphs that need to be loaded.
595             if (!queuedGlyphs.isEmpty()) displayList.invalid = true;
596         }
597 
598         GL.glTranslatef(-x, -y, 0);
599 
600         if (displayList == null) displayList = new DisplayList();
601         displayList.width = (short) maxWidth;
602         displayList.height = (short) (lines * getLineHeight() + totalHeight);
603         return displayList;
604     }

618     /**
619      * Returns the glyph for the specified codePoint. If the glyph does not exist yet,
620      * it is created and queued to be loaded.
621      *
622      * @param glyphCode The code of the glyph to locate
623      * @param codePoint The code point associated with the glyph
624      * @param bounds    The bounds of the glyph on the page
625      * @param vector    The vector the glyph is part of
626      * @param index     The index of the glyph within the vector
627      * @return The glyph requested
628      */
629     private Glyph getGlyph(int glyphCode, int codePoint, Rectangle bounds, GlyphVector vector, int index) {
630         if (glyphCode < 0 || glyphCode >= MAX_GLYPH_CODE) {
631             // GlyphVector#getGlyphCode sometimes returns negative numbers on OS X.
632             return new Glyph(codePoint, bounds, vector, index, this) {
633                 public boolean isMissing() {
634                     return true;
635                 }
636             };
637         }
638         int pageIndex = glyphCode / PAGE_SIZE;
639         int glyphIndex = glyphCode & (PAGE_SIZE - 1);
640         Glyph glyph = null;
641         Glyph[] page = glyphs[pageIndex];
642         if (page != null) {
643             glyph = page[glyphIndex];
644             if (glyph != null) return glyph;
645         } else
646             page = glyphs[pageIndex] = new Glyph[PAGE_SIZE];
647         // Add glyph so size information is available and queue it so its image can be loaded later.
648         glyph = page[glyphIndex] = new Glyph(codePoint, bounds, vector, index, this);
649         queuedGlyphs.add(glyph);
650         return glyph;
651     }

653     /**
654      * Returns the bounds of the specified glyph.\
655      *
656      * @param vector    The vector the glyph is part of
657      * @param index     The index of the glyph within the vector
658      * @param codePoint The code point associated with the glyph
659      */
660     private Rectangle getGlyphBounds(GlyphVector vector, int index, int codePoint) {

674      * @see org.newdawn.slick.Font#getWidth(java.lang.String)
675      */
676     public int getWidth(CharSequence text) {
677         if (text == null) throw new IllegalArgumentException("text cannot be null.");
678         if (text.length() == 0) return 0;
679 
680         if (displayListCaching) {
681             DisplayList displayList = (DisplayList) displayLists.get(text);
682             if (displayList != null) return displayList.width;
683         }
684 
685         char[] chars = toCharArray(text, 0, text.length());
686         GlyphVector vector = font.layoutGlyphVector(GlyphPage.renderContext, chars, 0, chars.length, Font.LAYOUT_LEFT_TO_RIGHT);
687 
688         int width = 0;
689         int extraX = 0;
690         boolean startNewLine = false;
691         for (int glyphIndex = 0, n = vector.getNumGlyphs(); glyphIndex < n; glyphIndex++) {
692             int charIndex = vector.getGlyphCharIndex(glyphIndex);
693             int codePoint = Character.codePointAt(text, charIndex);
694             Rectangle bounds = getGlyphBounds(vector, glyphIndex, codePoint);
695 
696             if (startNewLine && codePoint != '\n') extraX = -bounds.x;
697 
698             if (glyphIndex > 0) extraX += paddingLeft + paddingRight + paddingAdvanceX;
699             width = Math.max(width, bounds.x + extraX + bounds.width);
700 
701             if (codePoint == '\n') startNewLine = true;
702         }
703 
704         return width;
705     }

717         if (startIndex == 0 && endIndex == cs.length() && cs instanceof String)
718             return ((String) cs).toCharArray();

730     public int getHeight(CharSequence text) {
731         if (text == null) throw new IllegalArgumentException("text cannot be null.");
732         if (text.length() == 0) return 0;
733 
734         if (displayListCaching) {
735             DisplayList displayList = (DisplayList) displayLists.get(text);
736             if (displayList != null) return displayList.height;
737         }
738 
739         char[] chars = toCharArray(text, 0, text.length());
740         GlyphVector vector = font.layoutGlyphVector(GlyphPage.renderContext, chars, 0, chars.length, Font.LAYOUT_LEFT_TO_RIGHT);
741 
742         int lines = 0, height = 0;
743         for (int i = 0, n = vector.getNumGlyphs(); i < n; i++) {
744             int charIndex = vector.getGlyphCharIndex(i);
745             int codePoint = Character.codePointAt(text, charIndex);
746             if (codePoint == ' ') continue;
747             Rectangle bounds = getGlyphBounds(vector, i, codePoint);
748 
749             height = Math.max(height, ascent + bounds.y + bounds.height);

752                 lines++;
753                 height = 0;
754             }
755         }
756         return lines * getLineHeight() + height;
