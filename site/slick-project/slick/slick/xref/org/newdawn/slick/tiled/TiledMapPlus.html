107     /**
108      * Populates the objectGroupName to objectGroupOffset map and the
109      * groupObjectName to groupObjectOffset map
110      *
111      * @author liamzebedee
112      */
113     private void processNameToObjectMap() {
114         for (int i = 0; i < this.getObjectGroupCount(); i++) {
115             ObjectGroup g = this.objectGroups.get(i);
116             this.objectGroupNameToOffset.put(g.name, i);
117             HashMap<String, Integer> nameToObjectMap = new HashMap<String, Integer>();
118             for (int ib = 0; ib < this.getObjectCount(i); ib++) {
119                 nameToObjectMap.put(this.getObjectName(i, ib), ib);
120             }
121             g.setObjectNameMapping(nameToObjectMap);
122         }
123     }

180     /**
181      * Get all tiles from all layers that are part of a specific tileset
182      *
183      * @param tilesetName The name of the tileset that the tiles are part of
184      * @author liamzebedee
185      */
186     public ArrayList<Tile> getAllTilesFromAllLayers(String tilesetName) {
187         ArrayList<Tile> tiles = new ArrayList<Tile>();
188         int tilesetID = this.tilesetNameToIDMap.get(tilesetName);
189         for (int x = 0; x < this.getWidth(); x++) {
190             for (int y = 0; y < this.getHeight(); y++) {
191                 for (int l = 0; l < this.getLayerCount(); l++) {
192                     Layer layer = this.layers.get(l);
193                     if (layer.data[x][y][0] == tilesetID) {
194                         Tile t = new Tile(x, y, layer.name,
195                                 layer.data[x][y][2], tilesetName);
196                         tiles.add(t);
197                     }
198                 }
199             }
200         }
201         return tiles;
202     }

204     /**
205      * Writes the current TiledMap to a stream
206      * <br> Map is written using GZIP Base64 encoding in TiledMap V0.8
207      *
208      * @param stream The stream in which the TiledMap is to be written to
209      * @throws SlickException
210      * @author liamzebedee
211      */
212     public void write(OutputStream o) throws SlickException {
213         try {
214             DocumentBuilderFactory dbfac = DocumentBuilderFactory.newInstance();
215             DocumentBuilder docBuilder = dbfac.newDocumentBuilder();
216             Document doc = docBuilder.newDocument();
217 
218             Element map = doc.createElement("map");
219             map.setAttribute("version", "1.0");
220             if (this.orientation == TiledMap.ORTHOGONAL) map.setAttribute("orientation", "orthogonal");
221             else if (this.orientation == TiledMap.ISOMETRIC) map.setAttribute("orientation", "isometric");
222             map.setAttribute("tilewidth", "" + this.tileWidth);
223             map.setAttribute("tileheight", "" + this.tileHeight);
224             map.setAttribute("width", "" + this.width);
225             map.setAttribute("height", "" + this.height);
226             doc.appendChild(map);
227             for (int i = 0; i < this.tileSets.size(); i++) { // Loop through all tilesets
228                 TileSet tilesetData = this.tileSets.get(i);
229                 Element tileset = doc.createElement("tileset");
230                 tileset.setAttribute("firstgid", "" + tilesetData.firstGID);
231                 tileset.setAttribute("name", tilesetData.name);
232                 tileset.setAttribute("tilewidth", "" + tilesetData.tileWidth);
233                 tileset.setAttribute("tileheight", "" + tilesetData.tileHeight);
234                 tileset.setAttribute("spacing", "" + tilesetData.tileSpacing);
235                 tileset.setAttribute("margin", "" + tilesetData.tileMargin);
236                 Element image = doc.createElement("image");
237                 String imagePath = tilesetData.imageref.replaceFirst(
238                         this.getTilesLocation() + "/", "");
239                 image.setAttribute("source", imagePath);
240                 image.setAttribute("width", "" + tilesetData.tiles.getWidth());
241                 image.setAttribute("height", "" + tilesetData.tiles.getHeight());
242                 tileset.appendChild(image);
243                 int tileCount = tilesetData.tiles.getHorizontalCount()
244                         * tilesetData.tiles.getVerticalCount();
245                 Element tilesetProperties = doc.createElement("properties");
246                 Properties tilesetPropertiesData = tilesetData.tilesetProperties;
247                 if (tilesetProperties != null) {
248                     Enumeration propertyEnum = tilesetPropertiesData.propertyNames();
249                     while (propertyEnum.hasMoreElements()) {
250                         String key = (String) propertyEnum.nextElement();
251                         Element tileProperty = doc.createElement("property");
252                         tileProperty.setAttribute("name", key);
253                         tileProperty.setAttribute("value", tilesetPropertiesData.getProperty(key));
254                         tilesetProperties.appendChild(tileProperty);
255                     }
256                     tileset.appendChild(tilesetProperties);
257                 }
258                 if (tileCount == 1) {
259                     tileCount++;
260                 }
261                 for (int tileI = 0; tileI < tileCount; tileI++) {
262                     Properties tileProperties = tilesetData
263                             .getProperties(tileI);
264                     if (tileProperties != null) {
265                         Element tile = doc.createElement("tile");
266                         int tileID = tileI - tilesetData.firstGID;
267                         tile.setAttribute("id", "" + tileID);
268                         Element tileProps = doc.createElement("properties");
269 
270                         Enumeration propertyEnum = tilesetData.getProperties(
271                                 tileI).propertyNames();
272                         while (propertyEnum.hasMoreElements()) {
273                             String key = (String) propertyEnum.nextElement();
274                             Element tileProperty = doc
275                                     .createElement("property");
276                             tileProperty.setAttribute("name", key);
277                             tileProperty.setAttribute("value",
278                                     tileProperties.getProperty(key));
279                             tileProps.appendChild(tileProperty);
280                         }
281                         tile.appendChild(tileProps);
282                         tileset.appendChild(tile);
283                     }
284                 }
285                 map.appendChild(tileset);
286             }
287 
288             for (int i = 0; i < this.layers.size(); i++) {
289                 Element layer = doc.createElement("layer");
290                 Layer layerData = layers.get(i);
291                 layer.setAttribute("name", layerData.name);
292                 layer.setAttribute("width", "" + layerData.width);
293                 layer.setAttribute("height", "" + layerData.height);
294                 layer.setAttribute("opacity", "" + layerData.opacity);
295                 if (layerData.visible) layer.setAttribute("visible", "1");
296                 else layer.setAttribute("visible", "0");
297                 Element data = doc.createElement("data");
298 
299                 ByteArrayOutputStream os = new ByteArrayOutputStream();
300                 for (int tileY = 0; tileY < layerData.height; tileY++) {
301                     for (int tileX = 0; tileX < layerData.width; tileX++) {
302                         int tileGID = layerData.data[tileX][tileY][2];
303                         os.write(tileGID);
304                         os.write(tileGID << 8);
305                         os.write(tileGID << 16);
306                         os.write(tileGID << 24);
307                     }
308                 }
309                 os.flush();
310                 String compressedData = Base64.encodeBytes(os.toByteArray(),
311                         Base64.DONT_BREAK_LINES | Base64.GZIP | Base64.ENCODE);
312                 data.appendChild(doc.createTextNode(compressedData));
313                 data.setAttribute("encoding", "base64");
314                 data.setAttribute("compression", "gzip");
315 
316                 layer.appendChild(data);
317                 map.appendChild(layer);
318             }
319             for (int objectGroupI = 0; objectGroupI < this.objectGroups.size(); objectGroupI++) {
320                 Element objectGroup = doc.createElement("objectgroup");
321                 ObjectGroup objectGroupData = objectGroups.get(objectGroupI);
322                 objectGroup.setAttribute("color", "white");
323                 // It doesn't appear we use a color value,
324                 // but its in the format so...
325                 objectGroup.setAttribute("name", objectGroupData.name);
326                 objectGroup.setAttribute("width", "" + objectGroupData.width);
327                 objectGroup.setAttribute("height", "" + objectGroupData.height);
328                 objectGroup.setAttribute("opacity", "" + objectGroupData.opacity);
329                 if (objectGroupData.visible) objectGroup.setAttribute("visible", "1");
330                 else objectGroup.setAttribute("visible", "0");
331                 objectGroup.setAttribute("color", "#" +
332                         Float.toHexString(objectGroupData.color.r) +
333                         Float.toHexString(objectGroupData.color.g) +
334                         Float.toHexString(objectGroupData.color.b));
335 
336                 for (int groupObjectI = 0; groupObjectI < objectGroupData.objects
337                         .size(); groupObjectI++) {
338                     Element object = doc.createElement("object");
339                     GroupObject groupObject = objectGroupData.objects
340                             .get(groupObjectI);
341                     object.setAttribute("x", "" + groupObject.x);
342                     object.setAttribute("y", "" + groupObject.y);
343                     switch (groupObject.objectType) {
344                         case IMAGE:
345                             object.setAttribute("gid", "" + groupObject.gid);
346                             break;
347                         case RECTANGLE:
348                             object.setAttribute("name", groupObject.name);
349                             object.setAttribute("type", groupObject.type);
350                             object.setAttribute("width", "" + groupObject.width);
351                             object.setAttribute("height", "" + groupObject.height);
352                             break;
353                         case POLYGON:
354                             Element polygon = doc.createElement("polygon");
355                             String polygonPoints = "";
356                             for (int polygonPointIndex = 0; polygonPointIndex < groupObject.points.getPointCount() - 1; polygonPointIndex++) {
357                                 polygonPoints += groupObject.points.getPoint(polygonPointIndex)[0] + "," +
358                                         groupObject.points.getPoint(polygonPointIndex)[1] + " ";
359                             }
360                             polygonPoints.trim();
361                             polygon.setAttribute("points", polygonPoints);
362                             break;
363                         case POLYLINE:
364                             Element polyline = doc.createElement("polyline");
365                             String polylinePoints = "";
366                             for (int polyLinePointIndex = 0; polyLinePointIndex < groupObject.points.getPointCount() - 1; polyLinePointIndex++) {
367                                 polylinePoints += groupObject.points.getPoint(polyLinePointIndex)[0] + "," +
368                                         groupObject.points.getPoint(polyLinePointIndex)[1] + " ";
369                             }
370                             polylinePoints.trim();
371                             polyline.setAttribute("points", polylinePoints);
372                             break;
373                     }
374 
375                     if (groupObject.props != null) {
376                         Element objectProps = doc.createElement("properties");
377                         Enumeration propertyEnum = groupObject.props
378                                 .propertyNames();
379                         while (propertyEnum.hasMoreElements()) {
380                             String key = (String) propertyEnum.nextElement();
381                             Element objectProperty = doc
382                                     .createElement("property");
383                             objectProperty.setAttribute("name", key);
384                             objectProperty.setAttribute("value",
385                                     groupObject.props.getProperty(key));
386                             objectProps.appendChild(objectProperty);
387                         }
388                         object.appendChild(objectProps);
389                     }
390                     objectGroup.appendChild(object);
391 
392                 }
393 
394                 map.appendChild(objectGroup);
395             }
396             TransformerFactory transformerFactory = TransformerFactory
397                     .newInstance();
398             Transformer transformer = transformerFactory.newTransformer();
399             DOMSource source = new DOMSource(doc);
400             StreamResult result = new StreamResult(o);
401             transformer.transform(source, result);
402 
403         } catch (Exception e) {
404             Log.error(e);
405             throw new SlickException("Failed to write tiledmap", e);
406         }
407     }

427     /**
428      * Gets the visible tile's image at the co-ordinates
429      *
430      * @param x The x co-ordinate of the tile
431      * @param y The y co-ordinate of the tile
432      * @return The visible tile at this location
433      * @throws SlickException
434      * @author liamzebedee
435      */
436     public Image getVisibleTile(int x, int y) throws SlickException {
437         Image visibleTileImage = null;
438         for (int l = this.getLayerCount() - 1; l > -1; l--) {
439             if (visibleTileImage == null) {
440                 visibleTileImage = this.getTileImage(x, y, l);
441                 continue;
442             }
443         }
444         if (visibleTileImage == null) {
445             throw new SlickException("Tile doesn't have a tileset!");
446         }
447         return visibleTileImage;
448     }
