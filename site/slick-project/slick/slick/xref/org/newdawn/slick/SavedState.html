12  /**
13   * A utility to allow game setup/state to be stored locally. This utility will adapt to the
14   * current enviornment (webstart or file based). Note that this will not currently
15   * work in an applet.
16   *
17   * @author kappaOne
18   */
19  public class SavedState {
20      /**
21       * file name of where the scores will be saved
22       */
23      private String fileName;
24      /**
25       * Type of Muffin to use
26       */
27      private Muffin muffin;
28      /**
29       * hash map where int data will be stored
30       */
31      private HashMap numericData = new HashMap();
32      /**
33       * hash map where string data will be stored
34       */
35      private HashMap stringData = new HashMap();
36  
37      /**
38       * Create and Test to see if the app is running
39       * as webstart or local app and select the appropriate
40       * muffin type
41       *
42       * @param fileName name of muffin where data will be saved
43       * @throws SlickException Indicates a failure to load the stored state
44       */
45      public SavedState(String fileName) throws SlickException {
46          this.fileName = fileName;
47  
48          if (isWebstartAvailable()) {
49              muffin = new WebstartMuffin();
50          } else {
51              muffin = new FileMuffin();
52          }
53  
54          try {
55              load();
56          } catch (IOException e) {
57              throw new SlickException("Failed to load state on startup", e);
58          }
59      }

61      /**
62       * Get number stored at given location
63       *
64       * @param nameOfField The name of the number to retrieve
65       * @return The number saved at this location
66       */
67      public double getNumber(String nameOfField) {
68          return getNumber(nameOfField, 0);
69      }
70  
71      /**
72       * Get number stored at given location
73       *
74       * @param nameOfField  The name of the number to retrieve
75       * @param defaultValue The value to return if the specified value hasn't been set
76       * @return The number saved at this location
77       */
78      public double getNumber(String nameOfField, double defaultValue) {
79          Double value = ((Double) numericData.get(nameOfField));
80  
81          if (value == null) {
82              return defaultValue;
83          }
84  
85          return value.doubleValue();
86      }
87  
88      /**
89       * Save the given value at the given location
90       * will overwrite any previous value at this location
91       *
92       * @param nameOfField The name to store the value against
93       * @param value       The value to store
94       */
95      public void setNumber(String nameOfField, double value) {
96          numericData.put(nameOfField, new Double(value));
97      }
98  
99      /**
100      * Get the String at the given location
101      *
102      * @param nameOfField location of string
103      * @return String stored at the location given
104      */
105     public String getString(String nameOfField) {
106         return getString(nameOfField, null);
107     }
108 
109     /**
110      * Get the String at the given location
111      *
112      * @param nameOfField  location of string
113      * @param defaultValue The value to return if the specified value hasn't been set
114      * @return String stored at the location given
115      */
116     public String getString(String nameOfField, String defaultValue) {
117         String value = (String) stringData.get(nameOfField);
118 
119         if (value == null) {
120             return defaultValue;
121         }
122 
123         return value;
124     }
125 
126     /**
127      * Save the given value at the given location
128      * will overwrite any previous value at this location
129      *
130      * @param nameOfField location to store int
131      * @param value       The value to store
132      */
133     public void setString(String nameOfField, String value) {
134         stringData.put(nameOfField, value);
135     }

137     /**
138      * Save the stored data to file/muffin
139      *
140      * @throws IOException Indicates it wasn't possible to store the state
141      */
142     public void save() throws IOException {
143         muffin.saveFile(numericData, fileName + "_Number");
144         muffin.saveFile(stringData, fileName + "_String");
145     }
146 
147     /**
148      * Load the data from file/muffin
149      *
150      * @throws IOException Indicates it wasn't possible to load the state
151      */
152     public void load() throws IOException {
153         numericData = muffin.loadFile(fileName + "_Number");
154         stringData = muffin.loadFile(fileName + "_String");
155     }
156 
157     /**
158      * Will delete all current data held in Score
159      */
160     public void clear() {
161         numericData.clear();
162         stringData.clear();
163     }

165     /**
166      * Quick test to see if running through Java webstart
167      *
168      * @return True if jws running
169      */
170     private boolean isWebstartAvailable() {
171         try {
172             Class.forName("javax.jnlp.ServiceManager");
173             // this causes to go and see if the service is available
174             ServiceManager.lookup("javax.jnlp.PersistenceService");
175             Log.info("Webstart detected using Muffins");
176         } catch (Exception e) {
177             Log.info("Using Local File System");
178             return false;
179         }
180         return true;
181     }
