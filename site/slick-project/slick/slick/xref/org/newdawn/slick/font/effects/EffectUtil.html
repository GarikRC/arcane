127     /**
128      * Prompts the user for a value that represents a fixed number of options.
129      * All options are strings.
130      *
131      * @param options      The first array has an entry for each option. Each entry is either a String[1] that is both the display value
132      *                     and actual value, or a String[2] whose first element is the display value and second element is the actual value.
133      * @param name         The name of the value being prompted for
134      * @param currentValue The current value to show as default
135      * @param description  The description of the value
136      * @return The value selected by the user
137      */
138     static public Value optionValue(String name, final String currentValue, final String[][] options, final String description) {
139         return new DefaultValue(name, currentValue.toString()) {
140             public void showDialog() {
141                 int selectedIndex = -1;
142                 DefaultComboBoxModel model = new DefaultComboBoxModel();
143                 for (int i = 0; i < options.length; i++) {
144                     model.addElement(options[i][0]);
145                     if (getValue(i).equals(currentValue)) selectedIndex = i;
146                 }
147                 JComboBox comboBox = new JComboBox(model);
148                 comboBox.setSelectedIndex(selectedIndex);
149                 if (showValueDialog(comboBox, description)) value = getValue(comboBox.getSelectedIndex());
150             }
151 
152             private String getValue(int i) {
153                 if (options[i].length == 1) return options[i][0];
154                 return options[i][1];
155             }
156 
157             public String toString() {
158                 for (int i = 0; i < options.length; i++)
159                     if (getValue(i).equals(value)) return options[i][0].toString();
160                 return "";
161             }
162 
163             public Object getObject() {
164                 return value;
165             }
166         };
167     }

169     /**
170      * Convers a color to a string.
171      *
172      * @param color The color to encode to a string
173      * @return The colour as a string
174      */
175     static public String toString(Color color) {
176         if (color == null) throw new IllegalArgumentException("color cannot be null.");
177         String r = Integer.toHexString(color.getRed());
178         if (r.length() == 1) r = "0" + r;
179         String g = Integer.toHexString(color.getGreen());
180         if (g.length() == 1) g = "0" + g;
181         String b = Integer.toHexString(color.getBlue());
182         if (b.length() == 1) b = "0" + b;
183         return r + g + b;
184     }
185 
186     /**
187      * Converts a string to a color.
188      *
189      * @param rgb The string encoding the colour
190      * @return The colour represented by the given encoded string
191      */
192     static public Color fromString(String rgb) {
193         if (rgb == null || rgb.length() != 6) return Color.white;
194         return new Color(Integer.parseInt(rgb.substring(0, 2), 16), Integer.parseInt(rgb.substring(2, 4), 16), Integer.parseInt(rgb
195                 .substring(4, 6), 16));
196     }

261             ValueDialog dialog = new ValueDialog(component, name, description);

264             EventQueue.invokeLater(new Runnable() {
