79      /**
80       * Transform the point pairs in the source array and store them in the destination array.
81       * All operations will be done before storing the results in the destination.  This way the source
82       * and destination array can be the same without worry of overwriting information before it is transformed.
83       *
84       * @param source         Array of floats containing the points to be transformed
85       * @param sourceOffset   Where in the array to start processing
86       * @param destination    Array of floats to store the results.
87       * @param destOffset     Where in the array to start storing
88       * @param numberOfPoints Number of points to be transformed
89       * @throws ArrayIndexOutOfBoundsException if sourceOffset + numberOfPoints * 2 > source.length or the same operation on the destination array
90       */
91      public void transform(float source[], int sourceOffset, float destination[], int destOffset, int numberOfPoints) {
92          //TODO performance can be improved by removing the safety to the destination array   
93          float result[] = source == destination ? new float[numberOfPoints * 2] : destination;
94  
95          for (int i = 0; i < numberOfPoints * 2; i += 2) {
96              for (int j = 0; j < 6; j += 3) {
97                  result[i + (j / 3)] = source[i + sourceOffset] * matrixPosition[j] + source[i + sourceOffset + 1] * matrixPosition[j + 1] + 1 * matrixPosition[j + 2];
98              }
99          }
100 
101         if (source == destination) {
102             //for safety of the destination, the results are copied after the entire operation.
103             for (int i = 0; i < numberOfPoints * 2; i += 2) {
104                 destination[i + destOffset] = result[i];
105                 destination[i + destOffset + 1] = result[i + 1];
106             }
107         }
108     }

175     /**
176      * Create a new rotation Transform around the specified point
177      *
178      * @param angle The angle in radians to set the transform.
179      * @param x     The x coordinate around which to rotate.
180      * @param y     The y coordinate around which to rotate.
181      * @return The resulting Transform
182      */
183     public static Transform createRotateTransform(float angle, float x, float y) {
184         Transform temp = Transform.createRotateTransform(angle);
185         float sinAngle = temp.matrixPosition[3];
186         float oneMinusCosAngle = 1.0f - temp.matrixPosition[4];
187         temp.matrixPosition[2] = x * oneMinusCosAngle + y * sinAngle;
188         temp.matrixPosition[5] = y * oneMinusCosAngle - x * sinAngle;
189 
190         return temp;
191     }
