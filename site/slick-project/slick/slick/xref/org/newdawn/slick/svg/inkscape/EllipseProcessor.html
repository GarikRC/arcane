14   * Processor for <ellipse> and <path> nodes marked as arcs
15   *
16   * @author kevin
17   */
18  public class EllipseProcessor implements ElementProcessor {
19  
20      /**
21       * @see org.newdawn.slick.svg.inkscape.ElementProcessor#process(org.newdawn.slick.svg.Loader, org.w3c.dom.Element, org.newdawn.slick.svg.Diagram, org.newdawn.slick.geom.Transform)
22       */
23      public void process(Loader loader, Element element, Diagram diagram, Transform t) throws ParsingException {
24          Transform transform = Util.getTransform(element);
25          transform = new Transform(t, transform);
26  
27          float x = Util.getFloatAttribute(element, "cx");
28          float y = Util.getFloatAttribute(element, "cy");
29          float rx = Util.getFloatAttribute(element, "rx");
30          float ry = Util.getFloatAttribute(element, "ry");
31  
32          Ellipse ellipse = new Ellipse(x, y, rx, ry);
33          Shape shape = ellipse.transform(transform);
34  
35          NonGeometricData data = Util.getNonGeometricData(element);
36          data.addAttribute("cx", "" + x);
37          data.addAttribute("cy", "" + y);
38          data.addAttribute("rx", "" + rx);
39          data.addAttribute("ry", "" + ry);
40  
41          diagram.addFigure(new Figure(Figure.ELLIPSE, shape, data, transform));
42      }
43  
44      /**
45       * @see org.newdawn.slick.svg.inkscape.ElementProcessor#handles(org.w3c.dom.Element)
46       */
47      public boolean handles(Element element) {
48          if (element.getNodeName().equals("ellipse")) {
49              return true;
50          }
51          if (element.getNodeName().equals("path")) {
52              if ("arc".equals(element.getAttributeNS(Util.SODIPODI, "type"))) {
53                  return true;
54              }
55          }
56  
57          return false;
58      }
