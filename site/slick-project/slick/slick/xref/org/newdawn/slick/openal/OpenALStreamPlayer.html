157     /**
158      * Clean up the buffers applied to the sound source
159      */
160     private void removeBuffers() {
161         IntBuffer buffer = BufferUtils.createIntBuffer(1);
162         int queued = AL10.alGetSourcei(source, AL10.AL_BUFFERS_QUEUED);
163 
164         while (queued > 0) {
165             AL10.alSourceUnqueueBuffers(source, buffer);
166             queued--;

207     /**
208      * Poll the bufferNames - check if we need to fill the bufferNames with another
209      * section.
210      * <p/>
211      * Most of the time this should be reasonably quick
212      */
213     public void update() {
214         if (done) {
215             return;
216         }
217 
218         float sampleRate = audio.getRate();
219         float sampleSize;
220         if (audio.getChannels() > 1) {
221             sampleSize = 4; // AL10.AL_FORMAT_STEREO16
222         } else {
223             sampleSize = 2; // AL10.AL_FORMAT_MONO16
224         }
225 
226         int processed = AL10.alGetSourcei(source, AL10.AL_BUFFERS_PROCESSED);
227         while (processed > 0) {
228             unqueued.clear();
229             AL10.alSourceUnqueueBuffers(source, unqueued);
230 
231             int bufferIndex = unqueued.get(0);
232 
233             float bufferLength = (AL10.alGetBufferi(bufferIndex, AL10.AL_SIZE) / sampleSize) / sampleRate;
234             positionOffset += bufferLength;
235 
236             if (stream(bufferIndex)) {
237                 AL10.alSourceQueueBuffers(source, unqueued);
238             } else {
239                 remainingBufferCount--;
240                 if (remainingBufferCount == 0) {
241                     done = true;
242                 }
243             }
244             processed--;
245         }
246 
247         int state = AL10.alGetSourcei(source, AL10.AL_SOURCE_STATE);
248 
249         if (state != AL10.AL_PLAYING) {
250             AL10.alSourcePlay(source);
251         }
252     }

254     /**
255      * Stream some data from the audio stream to the buffer indicates by the ID
256      *
257      * @param bufferId The ID of the buffer to fill
258      * @return True if another section was available
259      */
260     public boolean stream(int bufferId) {
261         try {
262             int count = audio.read(buffer);
263 
264             if (count != -1) {
265                 bufferData.clear();
266                 bufferData.put(buffer, 0, count);
267                 bufferData.flip();
268 
269                 int format = audio.getChannels() > 1 ? AL10.AL_FORMAT_STEREO16 : AL10.AL_FORMAT_MONO16;
270                 try {
271                     AL10.alBufferData(bufferId, format, bufferData, audio.getRate());
272                 } catch (OpenALException e) {
273                     Log.error("Failed to loop buffer: " + bufferId + " " + format + " " + count + " " + audio.getRate(), e);
274                     return false;
275                 }
276             } else {
277                 if (loop) {
278                     initStreams();
279                     stream(bufferId);
280                 } else {
281                     done = true;
282                     return false;
283                 }
284             }
285 
286             return true;
287         } catch (IOException e) {
288             Log.error(e);
289             return false;
290         }
291     }

293     /**
294      * Seeks to a position in the music.
295      *
296      * @param position Position in seconds.
297      * @return True if the setting of the position was successful
298      */
299     public boolean setPosition(float position) {
300         try {
301             if (getPosition() > position) {
302                 initStreams();
303             }
304 
305             float sampleRate = audio.getRate();
306             float sampleSize;
307             if (audio.getChannels() > 1) {
308                 sampleSize = 4; // AL10.AL_FORMAT_STEREO16
309             } else {
310                 sampleSize = 2; // AL10.AL_FORMAT_MONO16
311             }
312 
313             while (positionOffset < position) {
314                 int count = audio.read(buffer);
315                 if (count != -1) {
316                     float bufferLength = (count / sampleSize) / sampleRate;
317                     positionOffset += bufferLength;
318                 } else {
319                     if (loop) {
320                         initStreams();
321                     } else {
322                         done = true;
323                     }
324                     return false;
325                 }
326             }
327 
328             startPlayback();
329 
330             return true;
331         } catch (IOException e) {
332             Log.error(e);
333             return false;
334         }
335     }

337     /**
338      * Starts the streaming.
339      */
340     private void startPlayback() {
341         AL10.alSourcei(source, AL10.AL_LOOPING, AL10.AL_FALSE);
342         AL10.alSourcef(source, AL10.AL_PITCH, pitch);
343 
344         remainingBufferCount = BUFFER_COUNT;
345 
346         for (int i = 0; i < BUFFER_COUNT; i++) {
347             stream(bufferNames.get(i));
348         }
349 
350         AL10.alSourceQueueBuffers(source, bufferNames);
351         AL10.alSourcePlay(source);
352     }
