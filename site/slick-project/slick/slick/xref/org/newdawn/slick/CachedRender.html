7   /**
8    * A set of rendering that is cached onto the graphics card and hopefully
9    * is quicker to render. Note that there are some things that can't be done
10   * in lists and that all dependent operations must be container. For instance,
11   * any colour configuration can not be assumed from outside the cache.
12   * <p/>
13   * Note: The destroy method needs to be used to tidy up. This is pretty important
14   * in this case since there are limited number of underlying resources.
15   *
16   * @author kevin
17   */
18  public class CachedRender {
19      /**
20       * The renderer to use for all GL operations
21       */
22      protected static SGL GL = Renderer.get();
23  
24      /**
25       * The operations to cache
26       */
27      private Runnable runnable;
28      /**
29       * The display list cached to
30       */
31      private int list = -1;
32  
33      /**
34       * Create a new cached render that will build the specified
35       * operations on to a video card resource
36       *
37       * @param runnable The operations to cache
38       */
39      public CachedRender(Runnable runnable) {
40          this.runnable = runnable;
41          build();
42      }
43  
44      /**
45       * Build the display list
46       */
47      private void build() {
48          if (list == -1) {
49              list = GL.glGenLists(1);
50  
51              SlickCallable.enterSafeBlock();
52              GL.glNewList(list, SGL.GL_COMPILE);
53              runnable.run();
54              GL.glEndList();
55              SlickCallable.leaveSafeBlock();
56          } else {
57              throw new RuntimeException("Attempt to build the display list more than once in CachedRender");
58          }
59      }
60  
61      /**
62       * Render the cached operations. Note that this doesn't call the operations, but
63       * rather calls the cached version
64       */
65      public void render() {
66          if (list == -1) {
67              throw new RuntimeException("Attempt to render cached operations that have been destroyed");
68          }
69  
70          SlickCallable.enterSafeBlock();
71          GL.glCallList(list);
72          SlickCallable.leaveSafeBlock();
73      }
74  
75      /**
76       * Destroy this cached render
77       */
78      public void destroy() {
79          GL.glDeleteLists(list, 1);
80          list = -1;
81      }
82  }
