64      /**
65       * Create a new group based on the XML definition
66       *
67       * @param element The XML element describing the layer
68       * @param map     The map to which the ObjectGroup belongs
69       * @throws SlickException Indicates a failure to parse the XML group
70       * @author kulpae
71       * @author liamzebedee
72       */
73      public ObjectGroup(Element element, TiledMap map) throws SlickException {
74          this.map = map;
75          TiledMapPlus tmap = null;
76          if (map instanceof TiledMapPlus) {
77              tmap = (TiledMapPlus) map;
78          }
79          name = element.getAttribute("name");
80          String widthS = element.getAttribute("width");
81          if (widthS != null && widthS.length() != 0) {
82              width = Integer.parseInt(widthS);
83          }
84          String heightS = element.getAttribute("height");
85          if (heightS != null && heightS.length() != 0) {
86              height = Integer.parseInt(heightS);
87          }
88          if (width == 0 || height == 0)
89              Log.warn("ObjectGroup " + name + " has zero size (width or height equal to 0)");
90  
91          objects = new ArrayList<GroupObject>();
92          String opacityS = element.getAttribute("opacity");
93          if (opacityS != null && opacityS.length() != 0) {
94              opacity = Float.parseFloat(opacityS);
95          }
96          if ("0".equals(element.getAttribute("visible"))) {
97              visible = false;
98          }
99          //will default to Color.white if attribute is not found / can't be parsed
100         String colorS = element.getAttribute("color");
101         if (colorS != null && colorS.length() != 0) {
102             try {
103                 color = Color.decode(colorS);
104             } catch (NumberFormatException e) {
105                 Log.warn("color attribute in element " + name + " could not be parsed; reverting to white");
106             }
107         }
108 
109         // now read the layer properties
110         Element propsElement = (Element) element.getElementsByTagName(
111                 "properties").item(0);
112         if (propsElement != null) {
113             NodeList properties = propsElement.getElementsByTagName("property");
114             if (properties != null) {
115                 props = new Properties();
116                 for (int p = 0; p < properties.getLength(); p++) {
117                     Element propElement = (Element) properties.item(p);
118                     String name = propElement.getAttribute("name");
119                     String value = propElement.getAttribute("value");
120                     props.setProperty(name, value);
121                 }
122             }
123         }
124 
125         NodeList objectNodes = element.getElementsByTagName("object");
126         for (int i = 0; i < objectNodes.getLength(); i++) {
127             Element objElement = (Element) objectNodes.item(i);
128             GroupObject object = null;
129             if (tmap != null) {
130                 object = new GroupObject(objElement, tmap);
131             } else {
132                 object = new GroupObject(objElement);
133             }
134             object.index = i;
135             objects.add(object);
136         }
137     }
