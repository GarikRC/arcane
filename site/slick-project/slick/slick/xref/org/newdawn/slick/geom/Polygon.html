20      /**
21       * Construct a new polygon with 3 or more points.
22       * This constructor will take the first set of points and copy them after
23       * the last set of points to create a closed shape.
24       *
25       * @param points An array of points in x, y order.
26       */
27      public Polygon(float points[]) {
28          int length = points.length;
29  
30          this.points = new float[length];
31          maxX = -Float.MIN_VALUE;
32          maxY = -Float.MIN_VALUE;
33          minX = Float.MAX_VALUE;
34          minY = Float.MAX_VALUE;
35          x = Float.MAX_VALUE;
36          y = Float.MAX_VALUE;
37  
38          for (int i = 0; i < length; i++) {
39              this.points[i] = points[i];
40              if (i % 2 == 0) {
41                  if (points[i] > maxX) {
42                      maxX = points[i];
43                  }
44                  if (points[i] < minX) {
45                      minX = points[i];
46                  }
47                  if (points[i] < x) {
48                      x = points[i];
49                  }
50              } else {
51                  if (points[i] > maxY) {
52                      maxY = points[i];
53                  }
54                  if (points[i] < minY) {
55                      minY = points[i];
56                  }
57                  if (points[i] < y) {
58                      y = points[i];
59                  }
60              }
61          }
62  
63          findCenter();
64          calculateRadius();
65          pointsDirty = true;
66      }

88      /**
89       * Add a point to the polygon
90       *
91       * @param x The x coordinate of the point
92       * @param y The y coordinate of the point
93       */
94      public void addPoint(float x, float y) {
95          if (hasVertex(x, y) && (!allowDups)) {
96              return;
97          }
98  
99          ArrayList tempPoints = new ArrayList();
100         for (int i = 0; i < points.length; i++) {
101             tempPoints.add(new Float(points[i]));
102         }
103         tempPoints.add(new Float(x));
104         tempPoints.add(new Float(y));
105         int length = tempPoints.size();
106         points = new float[length];
107         for (int i = 0; i < length; i++) {
108             points[i] = ((Float) tempPoints.get(i)).floatValue();
109         }
110         if (x > maxX) {
111             maxX = x;
112         }
113         if (y > maxY) {
114             maxY = y;
115         }
116         if (x < minX) {
117             minX = x;
118         }
119         if (y < minY) {
120             minY = y;
121         }
122         findCenter();
123         calculateRadius();
124 
125         pointsDirty = true;
126     }

131      * return the transformed shape.

136     public Shape transform(Transform transform) {
137         checkPoints();
138 
139         Polygon resultPolygon = new Polygon();
140 
141         float result[] = new float[points.length];
142         transform.transform(points, 0, result, 0, points.length / 2);
143         resultPolygon.points = result;
144         resultPolygon.findCenter();
145         resultPolygon.closed = closed;
146 
147         return resultPolygon;
