101     /**
102      * Add a defined point to the current contour
103      *
104      * @param pt The point to add
105      */
106     private void addPoint(Vector2f pt) {
107         if (holes == null) {
108             Point p = getPoint(pt);
109             contour.add(p);
110         } else {
111             Point p = getPoint(pt);
112             holes.add(p);
113         }
114     }

116     /**
117      * Triangulate the points given
118      *
119      * @param result The array to fill with the result or use to determine type
120      * @return The resultng triangles
121      */
122     private Vector2f[] triangulate(Vector2f[] result) {
123         // Step 1: Compute all angles
124         contour.computeAngles();
125         for (PointBag hole = holes; hole != null; hole = hole.next) {
126             hole.computeAngles();
127         }
128 
129         // Step 2: Connect the holes with the contour (build bridges)
130         while (holes != null) {
131             Point pHole = holes.first;
132             outer:
133             do {
134                 if (pHole.angle <= 0) {
135                     Point pContour = contour.first;
136                     do {
137                         inner:
138                         if (pHole.isInfront(pContour)
139                                 && pContour.isInfront(pHole)) {
140                             if (!contour.doesIntersectSegment(pHole.pt,
141                                     pContour.pt)) {
142                                 PointBag hole = holes;
143                                 do {
144                                     if (hole.doesIntersectSegment(pHole.pt,
145                                             pContour.pt)) {
146                                         break inner;
147                                     }
148                                 } while ((hole = hole.next) != null);
149 
150                                 Point newPtContour = getPoint(pContour.pt);
151                                 pContour.insertAfter(newPtContour);
152 
153                                 Point newPtHole = getPoint(pHole.pt);
154                                 pHole.insertBefore(newPtHole);
155 
156                                 pContour.next = pHole;
157                                 pHole.prev = pContour;
158 
159                                 newPtHole.next = newPtContour;
160                                 newPtContour.prev = newPtHole;
161 
162                                 pContour.computeAngle();
163                                 pHole.computeAngle();
164                                 newPtContour.computeAngle();
165                                 newPtHole.computeAngle();
166 
167                                 // detach the points from the hole
168                                 holes.first = null;
169                                 break outer;
170                             }
171                         }
172                     } while ((pContour = pContour.next) != contour.first);
173                 }
174             } while ((pHole = pHole.next) != holes.first);
175 
176             // free the hole
177             holes = freePointBag(holes);
178         }
179 
180         // Step 3: Make sure we have enough space for the result
181         int numTriangles = contour.countPoints() - 2;
182         int neededSpace = numTriangles * 3 + 1; // for the null
183         if (result.length < neededSpace) {
184             result = (Vector2f[]) Array.newInstance(result.getClass()
185                     .getComponentType(), neededSpace);
186         }
187 
188         // Step 4: Extract the triangles
189         int idx = 0;
190         for (; ; ) {
191             Point pContour = contour.first;
192 
193             if (pContour == null) {
194                 break;
195             }
196             // Are there 2 or less points left ?
197             if (pContour.next == pContour.prev) {
198                 break;
199             }
200 
201             outer:
202             do {
203                 if (pContour.angle > 0) {
204                     Point prev = pContour.prev;
205                     Point next = pContour.next;
206 
207                     if (next.next == prev || prev.isInfront(next) && next.isInfront(prev)) {
208                         if (!contour.doesIntersectSegment(prev.pt, next.pt)) {
209                             result[idx++] = pContour.pt;
210                             result[idx++] = next.pt;
211                             result[idx++] = prev.pt;
212                             break outer;
213                         }
214                     }
215                 }
216             } while ((pContour = pContour.next) != contour.first);
217 
218             // remove the point - we do it in every case to prevent endless loop
219             Point prev = pContour.prev;
220             Point next = pContour.next;
221 
222             contour.first = prev;
223             pContour.unlink();
224             freePoint(pContour);
225 
226             next.computeAngle();
227             prev.computeAngle();
228         }
229 
230         // Step 5: Append a null (see Collection.toArray)
231         result[idx] = null;
232 
233         // Step 6: Free memory
234         contour.clear();
235 
236         // Finished !
237         return result;
238     }

240     /**
241      * Create a new point bag (or recycle an old one)
242      *
243      * @return The new point bag
244      */
245     private PointBag getPointBag() {
246         PointBag pb = nextFreePointBag;
247         if (pb != null) {
248             nextFreePointBag = pb.next;
249             pb.next = null;
250             return pb;
251         }
252         return new PointBag();
253     }

255     /**
256      * Release a pooled bag
257      *
258      * @param pb The bag to release
259      * @return The next available bag
260      */
261     private PointBag freePointBag(PointBag pb) {
262         PointBag next = pb.next;
263         pb.clear();
264         pb.next = nextFreePointBag;
265         nextFreePointBag = pb;
266         return next;

269     /**
270      * Create or reuse a point
271      *
272      * @param pt The point data to set
273      * @return The new point
274      */
275     private Point getPoint(Vector2f pt) {
276         Point p = nextFreePoint;
277         if (p != null) {
278             nextFreePoint = p.next;
279             // initialize new point to safe values
280             p.next = null;
281             p.prev = null;
282             p.pt = pt;
283             return p;
284         }
285         return new Point(pt);
286     }

387         /**
388          * Java 5 hypot method
389          *
390          * @param x The x component
391          * @param y The y component
392          * @return The hypotenuse
393          */
394         private double hypot(double x, double y) {
395             return Math.sqrt(x * x + y * y);
396         }
397 
398         /**
399          * Compute the angle at this point
400          */
401         public void computeAngle() {
402             if (prev.pt.equals(pt)) {
403                 pt.x += 0.01f;
404             }
405             double dx1 = pt.x - prev.pt.x;
406             double dy1 = pt.y - prev.pt.y;
407             double len1 = hypot(dx1, dy1);
408             dx1 /= len1;
409             dy1 /= len1;
410 
411             if (next.pt.equals(pt)) {
412                 pt.y += 0.01f;
413             }
414             double dx2 = next.pt.x - pt.x;
415             double dy2 = next.pt.y - pt.y;
416             double len2 = hypot(dx2, dy2);
417             dx2 /= len2;
418             dy2 /= len2;
419 
420             double nx1 = -dy1;
421             double ny1 = dx1;
422 
423             nx = (nx1 - dy2) * 0.5;
424             ny = (ny1 + dx2) * 0.5;
425 
426             if (nx * nx + ny * ny < EPSILON) {
427                 nx = dx1;
428                 ny = dy2;
429                 angle = 1; // TODO: nx1,ny1 and nx2,ny2 facing ?
430                 if (dx1 * dx2 + dy1 * dy2 > 0) {
431                     nx = -dx1;
432                     ny = -dy1;
433                 }
434             } else {
435                 angle = nx * dx2 + ny * dy2;
436             }
437         }
438 
439         /**
440          * Get the angle of this point to another
441          *
442          * @param p The other point
443          * @return The angle between this point and another
444          */
445         public double getAngle(Point p) {
446             double dx = p.pt.x - pt.x;
447             double dy = p.pt.y - pt.y;
448             double dlen = hypot(dx, dy);
449 
450             return (nx * dx + ny * dy) / dlen;
451         }

453         /**
454          * Check if this point is convave
455          *
456          * @return True if this point remains concave
457          */
458         public boolean isConcave() {
459             return angle < 0;
460         }
461 
462         /**
463          * Check if this point is infront of another
464          *
465          * @param dx The other x
466          * @param dy The other y
467          * @return True if this point is infront (in the contour)
468          */
469         public boolean isInfront(double dx, double dy) {
470             // no nead to normalize, amplitude does not metter for side
471             // detection
472             boolean sidePrev = ((prev.pt.y - pt.y) * dx + (pt.x - prev.pt.x)
473                     * dy) >= 0;
474             boolean sideNext = ((pt.y - next.pt.y) * dx + (next.pt.x - pt.x)
475                     * dy) >= 0;
476 
477             return (angle < 0) ? (sidePrev | sideNext) : (sidePrev & sideNext);
478         }
479 
480         /**
481          * Check if this point is infront of another
482          *
483          * @param p The other point
484          * @return True if this point is infront (in the contour)
485          */
486         public boolean isInfront(Point p) {
487             return isInfront(p.pt.x - pt.x, p.pt.y - pt.y);
488         }

531         /**
532          * Compute the angles for the points in this bag
533          */
534         public void computeAngles() {
535             if (first == null) {
536                 return;
537             }
538 
539             Point p = first;
540             do {
541                 p.computeAngle();
542             } while ((p = p.next) != first);
543         }

545         /**
546          * Check if the points in this bag form a path intersecting
547          * with the specified path
548          *
549          * @param v1 The start point of the segment
550          * @param v2 The end point of the segment
551          * @return True if points in this contour intersect with the segment
552          */
553         public boolean doesIntersectSegment(Vector2f v1, Vector2f v2) {
554             double dxA = v2.x - v1.x;
555             double dyA = v2.y - v1.y;
556 
557             for (Point p = first; ; ) {
558                 Point n = p.next;
559                 if (p.pt != v1 && n.pt != v1 && p.pt != v2 && n.pt != v2) {
560                     double dxB = n.pt.x - p.pt.x;
561                     double dyB = n.pt.y - p.pt.y;
562                     double d = (dxA * dyB) - (dyA * dxB);
563 
564                     if (Math.abs(d) > EPSILON) {
565                         double tmp1 = p.pt.x - v1.x;
566                         double tmp2 = p.pt.y - v1.y;
567                         double tA = (dyB * tmp1 - dxB * tmp2) / d;
568                         double tB = (dyA * tmp1 - dxA * tmp2) / d;
569 
570                         if (tA >= 0 && tA <= 1 && tB >= 0 && tB <= 1) {
571                             return true;
572                         }
573                     }
574                 }
575 
576                 if (n == first) {
577                     return false;
578                 }
579                 p = n;
580             }
581         }

583         /**
584          * Get the number of points in the bag
585          *
586          * @return The number of points in the bag
587          */
588         public int countPoints() {
589             if (first == null) {
590                 return 0;
591             }
592 
593             int count = 0;
594             Point p = first;
595             do {
596                 ++count;
597             } while ((p = p.next) != first);
598             return count;
599         }

602          * Check if the point provided was contained
603          *
604          * @param point The point provided
605          * @return True if it's in the bag
606          */
607         public boolean contains(Vector2f point) {
608             if (first == null) {
609                 return false;
610             }
611 
612             if (first.prev.pt.equals(point)) {
613                 return true;
614             }
615             if (first.pt.equals(point)) {
616                 return true;
617             }
618             return false;
619         }

622     public boolean triangulate() {
623         Vector2f[] temp = triangulate(new Vector2f[0]);
624 
625         for (int i = 0; i < temp.length; i++) {
626             if (temp[i] == null) {
627                 break;
628             } else {
629                 triangles.add(temp[i]);
630             }
631         }
632 
633         return true;
634     }
