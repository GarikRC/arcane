91      /**
92       * Create a new set of configuration from a file
93       *
94       * @param in The stream from which to read the settings from
95       * @throws SlickException if the file could not be read.
96       */
97      public HieroSettings(InputStream in) throws SlickException {
98          try {
99              BufferedReader reader = new BufferedReader(new InputStreamReader(in));
100             while (true) {
101                 String line = reader.readLine();
102                 if (line == null) break;
103                 line = line.trim();
104                 if (line.length() == 0) continue;
105                 String[] pieces = line.split("=", 2);
106                 String name = pieces[0].trim();
107                 String value = pieces[1];
108                 if (name.equals("font.size")) {
109                     fontSize = Integer.parseInt(value);
110                 } else if (name.equals("font.bold")) {
111                     bold = Boolean.valueOf(value).booleanValue();
112                 } else if (name.equals("font.italic")) {
113                     italic = Boolean.valueOf(value).booleanValue();
114                 } else if (name.equals("pad.top")) {
115                     paddingTop = Integer.parseInt(value);
116                 } else if (name.equals("pad.right")) {
117                     paddingRight = Integer.parseInt(value);
118                 } else if (name.equals("pad.bottom")) {
119                     paddingBottom = Integer.parseInt(value);
120                 } else if (name.equals("pad.left")) {
121                     paddingLeft = Integer.parseInt(value);
122                 } else if (name.equals("pad.advance.x")) {
123                     paddingAdvanceX = Integer.parseInt(value);
124                 } else if (name.equals("pad.advance.y")) {
125                     paddingAdvanceY = Integer.parseInt(value);
126                 } else if (name.equals("glyph.page.width")) {
127                     glyphPageWidth = Integer.parseInt(value);
128                 } else if (name.equals("glyph.page.height")) {
129                     glyphPageHeight = Integer.parseInt(value);
130                 } else if (name.equals("effect.class")) {
131                     try {
132                         effects.add(Class.forName(value).newInstance());
133                     } catch (Exception ex) {
134                         throw new SlickException("Unable to create effect instance: " + value, ex);
135                     }
136                 } else if (name.startsWith("effect.")) {
137                     // Set an effect value on the last added effect.
138                     name = name.substring(7);
139                     ConfigurableEffect effect = (ConfigurableEffect) effects.get(effects.size() - 1);
140                     List values = effect.getValues();
141                     for (Iterator iter = values.iterator(); iter.hasNext(); ) {
142                         Value effectValue = (Value) iter.next();
143                         if (effectValue.getName().equals(name)) {
144                             effectValue.setString(value);
145                             break;
146                         }
147                     }
148                     effect.setValues(values);
149                 }
150             }
151             reader.close();
152         } catch (Exception ex) {
153             throw new SlickException("Unable to load Hiero font file", ex);
154         }
155     }
