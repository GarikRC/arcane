43      /**
44       * Build a navigation mesh based on a tile map
45       *
46       * @param map       The map to build the navigation mesh from
47       * @param tileBased True if we'll use the tiles for the mesh initially
48       *                  rather than quad spacing
49       * @return The newly created navigation mesh
50       */
51      public NavMesh build(TileBasedMap map, boolean tileBased) {
52          this.tileBased = tileBased;
53  
54          ArrayList spaces = new ArrayList();
55  
56          if (tileBased) {
57              for (int x = 0; x < map.getWidthInTiles(); x++) {
58                  for (int y = 0; y < map.getHeightInTiles(); y++) {
59                      if (!map.blocked(this, x, y)) {
60                          spaces.add(new Space(x, y, 1, 1));
61                      }
62                  }
63              }
64          } else {
65              Space space = new Space(0, 0, map.getWidthInTiles(), map.getHeightInTiles());
66  
67              subsection(map, space, spaces);
68          }
69  
70          while (mergeSpaces(spaces)) {
71          }
72          linkSpaces(spaces);
73  
74          return new NavMesh(spaces);
75      }

77      /**
78       * Merge the spaces that have been created to optimize out anywhere
79       * we can.
80       *
81       * @param spaces The list of spaces to be merged
82       * @return True if a merge occured and we'll have to start the merge
83       *         process again
84       */
85      private boolean mergeSpaces(ArrayList spaces) {
86          for (int source = 0; source < spaces.size(); source++) {
87              Space a = (Space) spaces.get(source);
88  
89              for (int target = source + 1; target < spaces.size(); target++) {
90                  Space b = (Space) spaces.get(target);
91  
92                  if (a.canMerge(b)) {
93                      spaces.remove(a);
94                      spaces.remove(b);
95                      spaces.add(a.merge(b));
96                      return true;
97                  }
98              }
99          }
100 
101         return false;
102     }

104     /**
105      * Determine the links between spaces
106      *
107      * @param spaces The spaces to link up
108      */
109     private void linkSpaces(ArrayList spaces) {
110         for (int source = 0; source < spaces.size(); source++) {
111             Space a = (Space) spaces.get(source);
112 
113             for (int target = source + 1; target < spaces.size(); target++) {
114                 Space b = (Space) spaces.get(target);
115 
116                 if (a.hasJoinedEdge(b)) {
117                     a.link(b);
118                     b.link(a);
119                 }
120             }
121         }
122     }

124     /**
125      * Check if a particular space is clear of blockages
126      *
127      * @param map   The map the spaces are being built from
128      * @param space The space to check
129      * @return True if there are no blockages in the space
130      */
131     public boolean clear(TileBasedMap map, Space space) {
132         if (tileBased) {
133             return true;
134         }
135 
136         float x = 0;
137         boolean donex = false;
138 
139         while (x < space.getWidth()) {
140             float y = 0;
141             boolean doney = false;
142 
143             while (y < space.getHeight()) {
144                 sx = (int) (space.getX() + x);
145                 sy = (int) (space.getY() + y);
146 
147                 if (map.blocked(this, sx, sy)) {
148                     return false;
149                 }
150 
151                 y += 0.1f;
152                 if ((y > space.getHeight()) && (!doney)) {
153                     y = space.getHeight();
154                     doney = true;
155                 }
156             }
157 
158 
159             x += 0.1f;
160             if ((x > space.getWidth()) && (!donex)) {
161                 x = space.getWidth();
162                 donex = true;
163             }
164         }
165 
166         return true;
167     }

169     /**
170      * Subsection a space into smaller spaces if required to find a non-blocked
171      * area.
172      *
173      * @param map    The map being processed
174      * @param space  The space being sections
175      * @param spaces The list of spaces that have been created
176      */
177     private void subsection(TileBasedMap map, Space space, ArrayList spaces) {
178         if (!clear(map, space)) {
179             float width2 = space.getWidth() / 2;
180             float height2 = space.getHeight() / 2;
181 
182             if ((width2 < smallestSpace) && (height2 < smallestSpace)) {
183                 return;
184             }
185 
186             subsection(map, new Space(space.getX(), space.getY(), width2, height2), spaces);
187             subsection(map, new Space(space.getX(), space.getY() + height2, width2, height2), spaces);
188             subsection(map, new Space(space.getX() + width2, space.getY(), width2, height2), spaces);
189             subsection(map, new Space(space.getX() + width2, space.getY() + height2, width2, height2), spaces);
190         } else {
191             spaces.add(space);
192         }
193     }
