50      /**
51       * Flip the endian-ness of the short
52       *
53       * @param signedShort The short to flip
54       * @return The flipped short
55       */
56      private short flipEndian(short signedShort) {
57          int input = signedShort & 0xFFFF;
58          return (short) (input << 8 | (input & 0xFF00) >>> 8);
59      }

110     /**
111      * @see org.newdawn.slick.opengl.LoadableImageData#loadImage(java.io.InputStream, boolean, boolean, int[])
112      */
113     public ByteBuffer loadImage(InputStream fis, boolean flipped, boolean forceAlpha, int[] transparent) throws IOException {
114         if (transparent != null) {
115             forceAlpha = true;
116         }
117         byte red = 0;
118         byte green = 0;
119         byte blue = 0;
120         byte alpha = 0;
121 
122         BufferedInputStream bis = new BufferedInputStream(fis, 100000);
123         DataInputStream dis = new DataInputStream(bis);
124 
125         // Read in the Header
126         short idLength = (short) dis.read();
127         short colorMapType = (short) dis.read();
128         short imageType = (short) dis.read();
129         short cMapStart = flipEndian(dis.readShort());
130         short cMapLength = flipEndian(dis.readShort());
131         short cMapDepth = (short) dis.read();
132         short xOffset = flipEndian(dis.readShort());
133         short yOffset = flipEndian(dis.readShort());
134 
135         if (imageType != 2) {
136             throw new IOException("Slick only supports uncompressed RGB(A) TGA images");
137         }
138 
139         width = flipEndian(dis.readShort());
140         height = flipEndian(dis.readShort());
141         short pixelDepth = (short) dis.read();
142         if (pixelDepth == 32) {
143             forceAlpha = false;
144         }
145 
146         texWidth = get2Fold(width);
147         texHeight = get2Fold(height);
148 
149         short imageDescriptor = (short) dis.read();
150         if ((imageDescriptor & 0x0020) == 0) {
151             flipped = !flipped;
152         }
153 
154         // Skip image ID
155         if (idLength > 0) {
156             bis.skip(idLength);
157         }
158 
159         byte[] rawData = null;
160         if ((pixelDepth == 32) || (forceAlpha)) {
161             pixelDepth = 32;
162             format = Format.RGBA;
163             rawData = new byte[texWidth * texHeight * 4];
164         } else if (pixelDepth == 24) {
165             format = Format.RGB;
166             rawData = new byte[texWidth * texHeight * 3];
167         } else {
168             throw new RuntimeException("Only 24 and 32 bit TGAs are supported");
169         }
170 
171         if (pixelDepth == 24) {
172             if (flipped) {
173                 for (int i = height - 1; i >= 0; i--) {
174                     for (int j = 0; j < width; j++) {
175                         blue = dis.readByte();
176                         green = dis.readByte();
177                         red = dis.readByte();
178 
179                         int ofs = ((j + (i * texWidth)) * 3);
180                         rawData[ofs] = red;
181                         rawData[ofs + 1] = green;
182                         rawData[ofs + 2] = blue;
183                     }
184                 }
185             } else {
186                 for (int i = 0; i < height; i++) {
187                     for (int j = 0; j < width; j++) {
188                         blue = dis.readByte();
189                         green = dis.readByte();
190                         red = dis.readByte();
191 
192                         int ofs = ((j + (i * texWidth)) * 3);
193                         rawData[ofs] = red;
194                         rawData[ofs + 1] = green;
195                         rawData[ofs + 2] = blue;
196                     }
197                 }
198             }
199         } else if (pixelDepth == 32) {
200             if (flipped) {
201                 for (int i = height - 1; i >= 0; i--) {
202                     for (int j = 0; j < width; j++) {
203                         blue = dis.readByte();
204                         green = dis.readByte();
205                         red = dis.readByte();
206                         if (forceAlpha) {
207                             alpha = (byte) 255;
208                         } else {
209                             alpha = dis.readByte();
210                         }
211 
212                         int ofs = ((j + (i * texWidth)) * 4);
213 
214                         rawData[ofs] = red;
215                         rawData[ofs + 1] = green;
216                         rawData[ofs + 2] = blue;
217                         rawData[ofs + 3] = alpha;
218 
219                         if (alpha == 0) {
220                             rawData[ofs + 2] = (byte) 0;
221                             rawData[ofs + 1] = (byte) 0;
222                             rawData[ofs] = (byte) 0;
223                         }
224                     }
225                 }
226             } else {
227                 for (int i = 0; i < height; i++) {
228                     for (int j = 0; j < width; j++) {
229                         blue = dis.readByte();
230                         green = dis.readByte();
231                         red = dis.readByte();
232                         if (forceAlpha) {
233                             alpha = (byte) 255;
234                         } else {
235                             alpha = dis.readByte();
236                         }
237 
238                         int ofs = ((j + (i * texWidth)) * 4);
239 
240                         if (ByteOrder.nativeOrder() == ByteOrder.BIG_ENDIAN) {
241                             rawData[ofs] = red;
242                             rawData[ofs + 1] = green;
243                             rawData[ofs + 2] = blue;
244                             rawData[ofs + 3] = alpha;
245                         } else {
246                             rawData[ofs] = red;
247                             rawData[ofs + 1] = green;
248                             rawData[ofs + 2] = blue;
249                             rawData[ofs + 3] = alpha;
250                         }
251 
252                         if (alpha == 0) {
253                             rawData[ofs + 2] = 0;
254                             rawData[ofs + 1] = 0;
255                             rawData[ofs] = 0;
256                         }
257                     }
258                 }
259             }
260         }
261         fis.close();
262 
263         if (transparent != null) {
264             for (int i = 0; i < rawData.length; i += 4) {
265                 boolean match = true;
266                 for (int c = 0; c < 3; c++) {
267                     if (rawData[i + c] != transparent[c]) {
268                         match = false;
269                     }
270                 }
271 
272                 if (match) {
273                     rawData[i + 3] = 0;
274                 }
275             }
276         }
277 
278         // Get a pointer to the image memory
279         ByteBuffer scratch = BufferUtils.createByteBuffer(rawData.length);
280         scratch.put(rawData);
281 
282         int perPixel = pixelDepth / 8;
283         if (height < texHeight - 1) {
284             int topOffset = (texHeight - 1) * (texWidth * perPixel);
285             int bottomOffset = (height - 1) * (texWidth * perPixel);
286             for (int x = 0; x < texWidth * perPixel; x++) {
287                 scratch.put(topOffset + x, scratch.get(x));
288                 scratch.put(bottomOffset + (texWidth * perPixel) + x, scratch.get((texWidth * perPixel) + x));
289             }
290         }
291         if (width < texWidth - 1) {
292             for (int y = 0; y < texHeight; y++) {
293                 for (int i = 0; i < perPixel; i++) {
294                     scratch.put(((y + 1) * (texWidth * perPixel)) - perPixel + i, scratch.get(y * (texWidth * perPixel) + i));
295                     scratch.put((y * (texWidth * perPixel)) + (width * perPixel) + i, scratch.get((y * (texWidth * perPixel)) + ((width - 1) * perPixel) + i));
296                 }
297             }
298         }
299 
300         scratch.flip();
301 
302         return scratch;
303     }

305     /**
306      * Get the closest greater power of 2 to the fold number
307      *
308      * @param fold The target number
309      * @return The power of 2
310      */
311     private int get2Fold(int fold) {
312         int ret = 2;
313         while (ret < fold) {
314             ret *= 2;
315         }
316         return ret;
317     }
