13  /**
14   * Provides a method of parsing XML into an existing data model. This does not
15   * provide the same functionality as JAXB or the variety of XML bindings out there. This
16   * is a utility to map XML onto an existing data model. The idea being that the design level
17   * model should not be driven by the XML schema thats defined. The two arn't always equal
18   * and often you end up with a set of class that represent your XML that you then have
19   * to traverse to extract into your normal data model.
20   * <p/>
21   * This utility hopes to take a piece of XML and map it onto a previously designed data
22   * model. At the moment it's way to tied to the structure of the XML but this will
23   * hopefully change with time.
24   * <p/>
25   * XML element names must be mapped to class names. This can be done in two ways either:
26   * <p/>
27   * - Specify an explict mapping with addElementMapping()
28   * - Specify the default package name and use the element name as the class name
29   * <p/>
30   * Each attribute in an element is mapped into a property of the element class, preferably
31   * through a set<AttrName> bean method, but alternatively by direct injection into private
32   * fields.
33   * <p/>
34   * Each child element is added to the target class by call the method add() on it with a single
35   * parameter of the type generated for the child element.
36   * <p/>
37   * Classes can optionally implement setXMLElementName(String) and setXMLElementContent(String) to
38   * recieve the name and content respectively of the XMLElement they were parsed from. This can
39   * help when mapping two elements to a single class.
40   * <p/>
41   * To reiterate, I'm not sure this is a good idea yet. It helps me as a utility since I've done
42   * this several times in the past but in the general case it may not be perfect. Consider a custom
43   * parser using XMLParser or JAXB (et al) seriously instead.
44   *
45   * @author kevin
46   */

217     /**
218      * Traverse the XML element specified generating the appropriate object structure
219      * for it and it's children
220      *
221      * @param current  The XML element to process
222      * @param instance The instance to parse onto, normally null
223      * @return The object created for the given element
224      * @throws SlickXMLException
225      */
226     private Object traverse(XMLElement current, Object instance) throws SlickXMLException {
227         String name = current.getName();
228         if (ignored.contains(name)) {
229             return null;
230         }
231 
232         Class clazz;
233 
234         if (instance == null) {
235             clazz = getClassForElementName(name);
236         } else {
237             clazz = instance.getClass();
238         }
239 
240         if (clazz == null) {
241             throw new SlickXMLException("Unable to map element " + name + " to a class, define the mapping");
242         }
243 
244         try {
245             if (instance == null) {
246                 instance = clazz.newInstance();
247 
248                 Method elementNameMethod = getMethod(clazz, "setXMLElementName", new Class[]{String.class});
249                 if (elementNameMethod != null) {
250                     invoke(elementNameMethod, instance, new Object[]{name});
251                 }
252                 Method contentMethod = getMethod(clazz, "setXMLElementContent", new Class[]{String.class});
253                 if (contentMethod != null) {
254                     invoke(contentMethod, instance, new Object[]{current.getContent()});
255                 }
256             }
257 
258             String[] attrs = current.getAttributeNames();
259             for (int i = 0; i < attrs.length; i++) {
260                 String methodName = "set" + attrs[i];
261 
262                 Method method = findMethod(clazz, methodName);
263 
264                 if (method == null) {
265                     Field field = findField(clazz, attrs[i]);
266                     if (field != null) {
267                         String value = current.getAttribute(attrs[i]);
268                         Object typedValue = typeValue(value, field.getType());
269                         setField(field, instance, typedValue);
270                     } else {
271                         Log.info("Unable to find property on: " + clazz + " for attribute: " + attrs[i]);
272                     }
273                 } else {
274                     String value = current.getAttribute(attrs[i]);
275                     Object typedValue = typeValue(value, method.getParameterTypes()[0]);
276                     invoke(method, instance, new Object[]{typedValue});
277                 }
278             }
279 
280             XMLElementList children = current.getChildren();
281             for (int i = 0; i < children.size(); i++) {
282                 XMLElement element = children.get(i);
283 
284                 Object child = traverse(element);
285                 if (child != null) {
286                     String methodName = addMethod;
287 
288                     Method method = findMethod(clazz, methodName, child.getClass());
289                     if (method == null) {
290                         Log.info("Unable to find method to add: " + child + " to " + clazz);
291                     } else {
292                         invoke(method, instance, new Object[]{child});
293                     }
294                 }
295             }
296 
297             return instance;
298         } catch (InstantiationException e) {
299             throw new SlickXMLException("Unable to instance " + clazz + " for element " + name + ", no zero parameter constructor?", e);
300         } catch (IllegalAccessException e) {
301             throw new SlickXMLException("Unable to instance " + clazz + " for element " + name + ", no zero parameter constructor?", e);
302         }
303     }

305     /**
306      * Convert a given value to a given type
307      *
308      * @param value The value to convert
309      * @param clazz The class that the returned object must be
310      * @return The value as the given type
311      * @throws SlickXMLException Indicates there is no automatic way of converting the value to the type
312      */
313     private Object typeValue(String value, Class clazz) throws SlickXMLException {
314         if (clazz == String.class) {
315             return value;
316         }
317 
318         try {
319             clazz = mapPrimitive(clazz);
320             return clazz.getConstructor(new Class[]{String.class}).newInstance(new Object[]{value});
321         } catch (Exception e) {
322             throw new SlickXMLException("Failed to convert: " + value + " to the expected primitive type: " + clazz, e);
323         }
324     }

326     /**
327      * Map a primitive class type to it's real object wrapper
328      *
329      * @param clazz The primitive type class
330      * @return The object wrapper class
331      */
332     private Class mapPrimitive(Class clazz) {
333         if (clazz == Integer.TYPE) {
334             return Integer.class;
335         }
336         if (clazz == Double.TYPE) {
337             return Double.class;
338         }
339         if (clazz == Float.TYPE) {
340             return Float.class;
341         }
342         if (clazz == Boolean.TYPE) {
343             return Boolean.class;
344         }
345         if (clazz == Long.TYPE) {
346             return Long.class;
347         }
348 
349         throw new RuntimeException("Unsupported primitive: " + clazz);
350     }

352     /**
353      * Find a field in a class by it's name. Note that this method is
354      * only needed because the general reflection method is case
355      * sensitive
356      *
357      * @param clazz The clazz to search
358      * @param name  The name of the field to search for
359      * @return The field or null if none could be located
360      */
361     private Field findField(Class clazz, String name) {
362         Field[] fields = clazz.getDeclaredFields();
363         for (int i = 0; i < fields.length; i++) {
364             if (fields[i].getName().equalsIgnoreCase(name)) {
365                 if (fields[i].getType().isPrimitive()) {
366                     return fields[i];
367                 }
368                 if (fields[i].getType() == String.class) {
369                     return fields[i];
370                 }
371             }
372         }
373 
374         return null;
375     }

377     /**
378      * Find a method in a class by it's name. Note that this method is
379      * only needed because the general reflection method is case
380      * sensitive
381      *
382      * @param clazz The clazz to search
383      * @param name  The name of the method to search for
384      * @return The method or null if none could be located
385      */
386     private Method findMethod(Class clazz, String name) {
387         Method[] methods = clazz.getDeclaredMethods();
388         for (int i = 0; i < methods.length; i++) {
389             if (methods[i].getName().equalsIgnoreCase(name)) {
390                 Method method = methods[i];
391                 Class[] params = method.getParameterTypes();
392 
393                 if (params.length == 1) {
394                     return method;
395                 }
396             }
397         }
398 
399         return null;
400     }

402     /**
403      * Find a method on a class with a single given parameter.
404      *
405      * @param clazz     The clazz to search through
406      * @param name      The name of the method to locate
407      * @param parameter The type the single parameter must have
408      * @return The method or null if none could be located
409      */
410     private Method findMethod(Class clazz, String name, Class parameter) {
411         Method[] methods = clazz.getDeclaredMethods();
412         for (int i = 0; i < methods.length; i++) {
413             if (methods[i].getName().equalsIgnoreCase(name)) {
414                 Method method = methods[i];
415                 Class[] params = method.getParameterTypes();
416 
417                 if (params.length == 1) {
418                     if (method.getParameterTypes()[0].isAssignableFrom(parameter)) {
419                         return method;
420                     }
421                 }
422             }
423         }
424 
425         return null;
426     }
