76          /**
77           * Create a new particle pool contiaining a set of particles
78           *
79           * @param system       The system that owns the particles over all
80           * @param maxParticles The maximum number of particles in the pool
81           */
82          public ParticlePool(ParticleSystem system, int maxParticles) {
83              particles = new Particle[maxParticles];
84              available = new ArrayList();
85  
86              for (int i = 0; i < particles.length; i++) {
87                  particles[i] = createParticle(system);
88              }
89  
90              reset(system);
91          }

93          /**
94           * Rest the list of particles
95           *
96           * @param system The system in which the particle belong
97           */
98          public void reset(ParticleSystem system) {
99              available.clear();
100 
101             for (int i = 0; i < particles.length; i++) {
102                 available.add(particles[i]);
103             }
104         }

418     /**
419      * Render the particles in the system
420      *
421      * @param x The x coordinate to render the particle system at (in the current coordinate space)
422      * @param y The y coordinate to render the particle system at (in the current coordiante space)
423      */
424     public void render(float x, float y) {
425         if ((sprite == null) && (defaultImageName != null)) {
426             loadSystemParticleImage();
427         }
428 
429         if (!visible) {
430             return;
431         }
432 
433         GL.glTranslatef(x, y, 0);
434 
435         if (blendingMode == BLEND_ADDITIVE) {
436             GL.glBlendFunc(SGL.GL_SRC_ALPHA, SGL.GL_ONE);
437         }
438         if (usePoints()) {
439             GL.glEnable(SGL.GL_POINT_SMOOTH);
440             TextureImpl.bindNone();
441         }
442 
443         // iterate over all emitters
444         for (int emitterIdx = 0; emitterIdx < emitters.size(); emitterIdx++) {
445             // get emitter
446             ParticleEmitter emitter = (ParticleEmitter) emitters.get(emitterIdx);
447 
448             if (!emitter.isEnabled()) {
449                 continue;
450             }
451 
452             // check for additive override and enable when set
453             if (emitter.useAdditive()) {
454                 GL.glBlendFunc(SGL.GL_SRC_ALPHA, SGL.GL_ONE);
455             }
456 
457             // now get the particle pool for this emitter and render all particles that are in use
458             ParticlePool pool = (ParticlePool) particlesByEmitter.get(emitter);
459             Image image = emitter.getImage();
460             if (image == null) {
461                 image = this.sprite;
462             }
463 
464             if (!emitter.isOriented() && !emitter.usePoints(this)) {
465                 image.startUse();
466             }
467 
468             for (int i = 0; i < pool.particles.length; i++) {
469                 if (pool.particles[i].inUse())
470                     pool.particles[i].render();
471             }
472 
473             if (!emitter.isOriented() && !emitter.usePoints(this)) {
474                 image.endUse();
475             }
476 
477             // reset additive blend mode
478             if (emitter.useAdditive()) {
479                 GL.glBlendFunc(SGL.GL_SRC_ALPHA, SGL.GL_ONE_MINUS_SRC_ALPHA);
480             }
481         }
482 
483         if (usePoints()) {
484             GL.glDisable(SGL.GL_POINT_SMOOTH);
485         }
486         if (blendingMode == BLEND_ADDITIVE) {
487             GL.glBlendFunc(SGL.GL_SRC_ALPHA, SGL.GL_ONE_MINUS_SRC_ALPHA);
488         }
489 
490         Color.white.bind();
491         GL.glTranslatef(-x, -y, 0);
492     }

515     /**
516      * Update the system, request the assigned emitters update the particles
517      *
518      * @param delta The amount of time thats passed since last update in milliseconds
519      */
520     public void update(int delta) {
521         if ((sprite == null) && (defaultImageName != null)) {
522             loadSystemParticleImage();
523         }
524 
525         removeMe.clear();
526         ArrayList emitters = new ArrayList(this.emitters);
527         for (int i = 0; i < emitters.size(); i++) {
528             ParticleEmitter emitter = (ParticleEmitter) emitters.get(i);
529             if (emitter.isEnabled()) {
530                 emitter.update(this, delta);
531                 if (removeCompletedEmitters) {
532                     if (emitter.completed()) {
533                         removeMe.add(emitter);
534                         particlesByEmitter.remove(emitter);
535                     }
536                 }
537             }
538         }
539         this.emitters.removeAll(removeMe);
540 
541         pCount = 0;
542 
543         if (!particlesByEmitter.isEmpty()) {
544             Iterator it = particlesByEmitter.keySet().iterator();
545             while (it.hasNext()) {
546                 ParticleEmitter emitter = (ParticleEmitter) it.next();
547                 if (emitter.isEnabled()) {
548                     ParticlePool pool = (ParticlePool) particlesByEmitter.get(emitter);
549                     for (int i = 0; i < pool.particles.length; i++) {
550                         if (pool.particles[i].life > 0) {
551                             pool.particles[i].update(delta);
552                             pCount++;
553                         }
554                     }
555                 }
556             }
557         }
558     }

570      * Get a new particle from the system. This should be used by emitters to

572      *
573      * @param emitter The emitter requesting the particle
574      * @param life    The time the new particle should live for
575      * @return A particle from the system
576      */
577     public Particle getNewParticle(ParticleEmitter emitter, float life) {
578         ParticlePool pool = (ParticlePool) particlesByEmitter.get(emitter);
579         ArrayList available = pool.available;
580         if (available.size() > 0) {
581             Particle p = (Particle) available.remove(available.size() - 1);
582             p.init(emitter, life);
583             p.setImage(sprite);
584 
585             return p;
586         }
587 
588         Log.warn("Ran out of particles (increase the limit)!");
589         return dummy;
590     }

604     /**
605      * Release all the particles owned by the specified emitter
606      *
607      * @param emitter The emitter owning the particles that should be released
608      */
609     public void releaseAll(ParticleEmitter emitter) {
610         if (!particlesByEmitter.isEmpty()) {
611             Iterator it = particlesByEmitter.values().iterator();
612             while (it.hasNext()) {
613                 ParticlePool pool = (ParticlePool) it.next();
614                 for (int i = 0; i < pool.particles.length; i++) {
615                     if (pool.particles[i].inUse()) {
616                         if (pool.particles[i].getEmitter() == emitter) {
617                             pool.particles[i].setLife(-1);
618                             release(pool.particles[i]);
619                         }
620                     }
621                 }
622             }
623         }
624     }

642     /**
643      * Create a duplicate of this system. This would have been nicer as a different interface
644      * but may cause to much API change headache. Maybe next full version release it should be
645      * rethought.
646      * <p/>
647      * TODO: Consider refactor at next point release
648      *
649      * @return A copy of this particle system
650      * @throws SlickException Indicates a failure during copy or a invalid particle system to be duplicated
651      */
652     public ParticleSystem duplicate() throws SlickException {
653         for (int i = 0; i < emitters.size(); i++) {
654             if (!(emitters.get(i) instanceof ConfigurableEmitter)) {
655                 throw new SlickException("Only systems contianing configurable emitters can be duplicated");
656             }
657         }
658 
659         ParticleSystem theCopy = null;
660         try {
661             ByteArrayOutputStream bout = new ByteArrayOutputStream();
662             ParticleIO.saveConfiguredSystem(bout, this);
663             ByteArrayInputStream bin = new ByteArrayInputStream(bout.toByteArray());
664             theCopy = ParticleIO.loadConfiguredSystem(bin);
665         } catch (IOException e) {
666             Log.error("Failed to duplicate particle system");
667             throw new SlickException("Unable to duplicated particle system", e);
668         }
669 
670         return theCopy;
671     }
