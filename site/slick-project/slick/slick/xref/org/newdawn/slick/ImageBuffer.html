113     /**
114      * Set a pixel in the image buffer
115      *
116      * @param x The x position of the pixel to set
117      * @param y The y position of the pixel to set
118      * @param r The red component to set (0->255)
119      * @param g The green component to set (0->255)
120      * @param b The blue component to set (0->255)
121      * @param a The alpha component to set (0->255)
122      */
123     public void setRGBA(int x, int y, int r, int g, int b, int a) {
124         if ((x < 0) || (x >= width) || (y < 0) || (y >= height)) {
125             throw new RuntimeException("Specified location: " + x + "," + y + " outside of image");
126         }
127 
128         int ofs = ((x + (y * texWidth)) * 4);
129 
130         if (ByteOrder.nativeOrder() == ByteOrder.BIG_ENDIAN) {
131             rawData[ofs] = (byte) b;
132             rawData[ofs + 1] = (byte) g;
133             rawData[ofs + 2] = (byte) r;
134             rawData[ofs + 3] = (byte) a;
135         } else {
136             rawData[ofs] = (byte) r;
137             rawData[ofs + 1] = (byte) g;
138             rawData[ofs + 2] = (byte) b;
139             rawData[ofs + 3] = (byte) a;
140         }
141     }

162     /**
163      * Get the closest greater power of 2 to the fold number
164      *

166      * @return The power of 2

168     private int get2Fold(int fold) {
