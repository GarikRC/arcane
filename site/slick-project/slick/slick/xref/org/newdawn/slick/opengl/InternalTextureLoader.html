308     /**
309      * Get a texture from a image file
310      *
311      * @param in           The stream from which we can load the image
312      * @param resourceName The name to give this image in the internal cache
313      * @param flipped      True if we should flip the image on the y-axis while loading
314      * @param filter       The filter to use when scaling the texture
315      * @param transparent  The colour to interpret as transparent or null if none
316      * @return The texture loaded
317      * @throws IOException Indicates a failure to load the image
318      */
319     public TextureImpl getTexture(InputStream in, String resourceName, boolean flipped, int filter, int[] transparent) throws IOException {
320         if (deferred) {
321             return new DeferredTexture(in, resourceName, flipped, filter, transparent);
322         }
323 
324         HashMap hash = texturesLinear;
325         if (filter == SGL.GL_NEAREST) {
326             hash = texturesNearest;
327         }
328 
329         String resName = resourceName;
330         if (transparent != null) {
331             resName += ":" + transparent[0] + ":" + transparent[1] + ":" + transparent[2];
332         }
333         resName += ":" + flipped;
334 
335         if (holdTextureData) {
336             TextureImpl tex = (TextureImpl) hash.get(resName);
337             if (tex != null) {
338                 return tex;
339             }
340         } else {
341             SoftReference ref = (SoftReference) hash.get(resName);
342             if (ref != null) {
343                 TextureImpl tex = (TextureImpl) ref.get();
344                 if (tex != null) {
345                     return tex;
346                 } else {
347                     hash.remove(resName);
348                 }
349             }
350         }
351 
352         // horrible test until I can find something more suitable
353         try {
354             GL.glGetError();
355         } catch (NullPointerException e) {
356             throw new RuntimeException("Image based resources must be loaded as part of init() or the game loop. They cannot be loaded before initialisation.");
357         }
358 
359         TextureImpl tex = getTexture(in, resourceName,
360                 SGL.GL_TEXTURE_2D,
361                 filter, filter, flipped, transparent);
362 
363         tex.setCacheName(resName);
364         if (holdTextureData) {
365             hash.put(resName, tex);
366         } else {
367             hash.put(resName, new SoftReference(tex));
368         }
369 
370         return tex;
371     }

373     private TextureImpl getTexture(InputStream in, String resourceName,
374                                    int target, int minFilter, int magFilter,
375                                    boolean flipped, int[] transparent) throws IOException {
376         // create the texture ID for this texture
377         ByteBuffer textureBuffer;
378 
379         LoadableImageData imageData = ImageDataFactory.getImageDataFor(resourceName);
380         textureBuffer = imageData.loadImage(new BufferedInputStream(in), flipped, transparent);
381 
382         int textureID = createTextureID();
383         TextureImpl texture = new TextureImpl(resourceName, target, textureID);
384         // bind this texture 
385         GL.glEnable(target);
386         GL.glBindTexture(target, textureID);
387 
388         int width;
389         int height;
390         int texWidth;
391         int texHeight;
392 
393         ImageData.Format format;
394 
395         width = imageData.getWidth();
396         height = imageData.getHeight();
397         format = imageData.getFormat();
398 
399         texture.setTextureWidth(imageData.getTexWidth());
400         texture.setTextureHeight(imageData.getTexHeight());
401 
402         texWidth = texture.getTextureWidth();
403         texHeight = texture.getTextureHeight();
404 
405         IntBuffer temp = BufferUtils.createIntBuffer(16);
406         GL.glGetInteger(SGL.GL_MAX_TEXTURE_SIZE, temp);
407         int max = temp.get(0);
408         if ((texWidth > max) || (texHeight > max)) {
409             throw new IOException("Attempt to allocate a texture to big for the current hardware");
410         }
411 
412         int srcPixelFormat = format.getOGLType();
413         int componentCount = format.getColorComponents();
414 
415         texture.setWidth(width);
416         texture.setHeight(height);
417         texture.setImageFormat(format);
418 
419         if (holdTextureData) {
420             texture.setTextureData(srcPixelFormat, componentCount, minFilter, magFilter, textureBuffer);
421         }
422 
423         GL.glTexParameteri(target, SGL.GL_TEXTURE_MIN_FILTER, minFilter);
424         GL.glTexParameteri(target, SGL.GL_TEXTURE_MAG_FILTER, magFilter);
425 
426         // produce a texture from the byte buffer
427         GL.glTexImage2D(target,
428                 0,
429                 dstPixelFormat,
430                 get2Fold(width),
431                 get2Fold(height),
432                 0,
433                 srcPixelFormat,
434                 SGL.GL_UNSIGNED_BYTE,
435                 textureBuffer);
436         return texture;
437     }

439     /**
440      * An advanced texture loading method providing more parameters for glTexImage2D.
441      * The created texture will not be placed in the cache.
442      * <p/>
443      * If genMipmaps is true, the loader will attempt to automatically build mipmaps
444      * with either GL30.glGenerateMipmap() or GL14.GL_GENERATE_MIPMAP. If the GL version
445      * is less than 1.4, then no mipmaps will be built and instead the magFilter (one
446      * of GL_LINEAR or GL_NEAREST) will be used for both minification and magnification.
447      * Users can determine mipmap generation support with isGenerateMipmapSupported().
448      * <p/>
449      * If the internalFormat is not null, then that will override the default pixel
450      * format described by this InternalTextureLoader (either GL_RGBA16 or GL_RGBA8
451      * depending on the is16BitMode() value). This parameter can be independent
452      * of the format of ImageData -- OpenGL will convert the ImageData format (e.g. BGRA)
453      * to the given internalFormat (e.g. RGB). Note that internalFormat is more limited
454      * than the ImageData's format; i.e. BGRA as an internal storage format is only
455      * supported if GL_ext_bgra is present.
456      * <p/>
457      * After calling this, the texture will be bound and the target (e.g. GL_TEXTURE_2D)
458      * will be enabled. If you are using a higher priority target, such as 3D textures,
459      * you should disable that afterwards to ensure compatibility with Slick.
460      * <p/>
461      * The ByteBuffer data is assumed to match getTexWidth/getTexHeight in ImageData.
462      *
463      * @param data           the image data holding width, height, format (ImageData byte buffer is ignored)
464      * @param buffer         the actual data to send to GL
465      * @param ref            The name to give the TextureImpl
466      * @param target         The texture target we're loading this texture into
467      * @param minFilter      The scaling down filter
468      * @param magFilter      The scaling up filter
469      * @param genMipmaps     true to generate mipmaps (failure will fallback to using magFilter)
470      * @param internalFormat the internal format of the texture (or null for default)
471      * @return The texture loaded
472      * @throws IOException Indicates a failure to load the image
473      */
474     public TextureImpl createTexture(ImageData data, ByteBuffer buffer,
475                                      String ref,
476                                      int target,
477                                      int minFilter,
478                                      int magFilter,
479                                      boolean genMipmaps,
480                                      ImageData.Format internalFormat) throws IOException {
481         int textureID = createTextureID();
482         TextureImpl texture = new TextureImpl(ref, target, textureID);
483         // bind this texture 
484         GL.glEnable(target);
485         GL.glBindTexture(target, textureID);
486 
487         int width = data.getWidth();
488         int height = data.getHeight();
489         int texWidth = data.getTexWidth();
490         int texHeight = data.getTexHeight();
491 
492         boolean usePOT = !isNPOTSupported() || isForcePOT();
493         if (usePOT) {
494             texWidth = get2Fold(width);
495             texHeight = get2Fold(height);
496         }
497 
498         int max = GL11.glGetInteger(SGL.GL_MAX_TEXTURE_SIZE);
499         if (texWidth > max || texHeight > max)
500             throw new IOException("Attempt to allocate a texture to big for the current hardware");
501 
502         ImageData.Format dataFormat = data.getFormat();
503         int dstFmt = internalFormat != null ? internalFormat.getOGLType() : dstPixelFormat;
504         int srcFmt = dataFormat.getOGLType();
505 
506         texture.setTextureWidth(texWidth);
507         texture.setTextureHeight(texHeight);
508         texture.setWidth(width);
509         texture.setHeight(height);
510         //even though it might really be RGBA16/8, user will expect comparability with Format constants
511         texture.setImageFormat(internalFormat != null ? internalFormat : ImageData.Format.RGBA);
512 
513         if (holdTextureData) {
514             // TODO: fix the reload functionality; right now it causes problems and
515             // should probably just be removed or reworked
516             int componentCount = dataFormat.getColorComponents();
517             texture.setTextureData(srcFmt, componentCount, minFilter, magFilter, buffer);
518         }
519 
520         ContextCapabilities cx = GLContext.getCapabilities();
521         if (genMipmaps && !isGenerateMipmapSupported()) { //nothing for auto mipmap gen
522             minFilter = magFilter;
523             genMipmaps = false;
524         }
525 
526         GL.glTexParameteri(target, SGL.GL_TEXTURE_MIN_FILTER, minFilter);
527         GL.glTexParameteri(target, SGL.GL_TEXTURE_MAG_FILTER, magFilter);
528 
529         //if we are < 3.0 and have no FBO support, fall back to GL_GENERATE_MIPMAP
530         if (genMipmaps && !cx.OpenGL30 && !cx.GL_EXT_framebuffer_object) {
531             GL.glTexParameteri(target, GL14.GL_GENERATE_MIPMAP, GL11.GL_TRUE);
532             genMipmaps = false;
533         }
534 
535         //For now, just assume Slick has decoded image data into POT
536         GL.glTexImage2D(target, 0, dstFmt, texWidth, texHeight, 0, srcFmt, SGL.GL_UNSIGNED_BYTE, buffer);
537 
538 //        if (texWidth==width && texHeight==height) {
539 //        	GL.glTexImage2D(target, 0, dstFmt, texWidth, texHeight, 
540 //        			0, srcFmt, SGL.GL_UNSIGNED_BYTE, buffer);
541 //        } else {
542 //        	//Slick2D decodes NPOT image data into padded byte buffers.
543 //        	//Once we make the shift to decoding NPOT image data, then we can clean this up
544 //        	GL.glTexImage2D(target, 0, dstFmt, texWidth, texHeight, 
545 //        			0, srcFmt, SGL.GL_UNSIGNED_BYTE, buffer);
546 //        	
547 //        	//first create the full texture
548 //        	//we could also use a null ByteBuffer but this seems to be buggy with certain machines
549 ////        	ByteBuffer empty = BufferUtils.createByteBuffer(texWidth * texHeight * 4);
550 ////        	GL.glTexImage2D(target, 0, dstFmt, texWidth, texHeight,
551 ////        			0, SGL.GL_RGBA, SGL.GL_UNSIGNED_BYTE, empty);
552 ////        	//then upload the sub image
553 ////        	GL.glTexSubImage2D(target, 0, 0, 0, width, height, srcFmt, SGL.GL_UNSIGNED_BYTE, buffer);
554 //        }
555 
556         if (genMipmaps) {
557             GL11.glEnable(target); //fixes ATI bug
558             if (cx.OpenGL30)
559                 GL30.glGenerateMipmap(target);
560             else
561                 EXTFramebufferObject.glGenerateMipmapEXT(target);
562         }
563         return texture;
564     }

669     /**
670      * Get the closest greater power of 2 to the fold number
671      *
672      * @param fold The target number
673      * @return The power of 2
674      */
675     public static int get2Fold(int fold) {
676         //new algorithm? -> return 1 << (32 - Integer.numberOfLeadingZeros(n-1));
677         int ret = 2;
678         while (ret < fold) {
679             ret *= 2;
680         }
681         return ret;
