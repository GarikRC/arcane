56      /**
57       * Start the game container rendering
58       *
59       * @throws SlickException Indicates a failure during game execution
60       */
61      public void start() throws SlickException {
62          destroyed = false;
63          SwingUtilities.invokeLater(new Runnable() {
64              public void run() {
65                  try {
66                      Input.disableControllers();
67  
68                      try {
69                          Display.setParent(CanvasGameContainer.this);
70                      } catch (LWJGLException e) {
71                          throw new SlickException("Failed to setParent of canvas", e);
72                      }
73  
74                      container.setup();
75                      scheduleUpdate();
76                  } catch (SlickException e) {
77                      e.printStackTrace();
78                      System.exit(0);
79                  }
80              }
81          });
82      }

84      /**
85       * Schedule an update on the EDT
86       */
87      private void scheduleUpdate() {
88          if (destroyed || !isVisible()) {
89              return;
90          }
91  
92          if (!container.running()) {
93              container.destroy();
94              destroyed = true;
95              return;
96          }
97  
98          SwingUtilities.invokeLater(new Runnable() {
99              public void run() {
100                 if (destroyed)
101                     return;
102                 try {
103                     container.gameLoop();
104                 } catch (SlickException e) {
105                     e.printStackTrace();
106                 }
107                 container.checkDimensions();
108                 scheduleUpdate();
109             }
110         });
111     }
