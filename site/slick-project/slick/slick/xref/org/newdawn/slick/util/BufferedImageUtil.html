64      /**
65       * Load a texture into OpenGL from a BufferedImage
66       *
67       * @param resourceName   The location of the resource to load
68       * @param resourceimage  The BufferedImage we are converting
69       * @param target         The GL target to load the texture against
70       * @param dstPixelFormat The pixel format of the screen (internalFormat for glTexImage)
71       * @param minFilter      The minimising filter
72       * @param magFilter      The magnification filter
73       * @return The loaded texture
74       * @throws IOException Indicates a failure to access the resource
75       */
76      public static Texture getTexture(String resourceName,
77                                       BufferedImage resourceimage, int target, int dstPixelFormat,
78                                       int minFilter, int magFilter) throws IOException {
79          ImageIOImageData data = new ImageIOImageData();
80          int srcPixelFormat = 0;
81  
82          // create the texture ID for this texture
83          int textureID = InternalTextureLoader.createTextureID();
84          TextureImpl.unbind();
85          TextureImpl texture = new TextureImpl(resourceName, target, textureID);
86  
87          // Enable texturing
88          Renderer.get().glEnable(SGL.GL_TEXTURE_2D);
89  
90          // bind this texture
91          Renderer.get().glBindTexture(target, textureID);
92  
93          BufferedImage bufferedImage = resourceimage;
94          texture.setWidth(bufferedImage.getWidth());
95          texture.setHeight(bufferedImage.getHeight());
96  
97          // convert that image into a byte buffer of texture data
98          ByteBuffer textureBuffer = data.imageToByteBuffer(bufferedImage, false, false, null);
99          texture.setTextureHeight(data.getTexHeight());
100         texture.setTextureWidth(data.getTexWidth());
101         texture.setImageFormat(data.getFormat());
102 
103         srcPixelFormat = data.getFormat().getOGLType();
104 
105         if (target == SGL.GL_TEXTURE_2D) {
106             Renderer.get().glTexParameteri(target, SGL.GL_TEXTURE_MIN_FILTER, minFilter);
107             Renderer.get().glTexParameteri(target, SGL.GL_TEXTURE_MAG_FILTER, magFilter);
108 
109             if (Renderer.get().canTextureMirrorClamp()) {
110                 Renderer.get().glTexParameteri(SGL.GL_TEXTURE_2D, SGL.GL_TEXTURE_WRAP_S, SGL.GL_MIRROR_CLAMP_TO_EDGE_EXT);
111                 Renderer.get().glTexParameteri(SGL.GL_TEXTURE_2D, SGL.GL_TEXTURE_WRAP_T, SGL.GL_MIRROR_CLAMP_TO_EDGE_EXT);
112             } else {
113                 Renderer.get().glTexParameteri(SGL.GL_TEXTURE_2D, SGL.GL_TEXTURE_WRAP_S, SGL.GL_CLAMP);
114                 Renderer.get().glTexParameteri(SGL.GL_TEXTURE_2D, SGL.GL_TEXTURE_WRAP_T, SGL.GL_CLAMP);
115             }
116         }
117 
118         Renderer.get().glTexImage2D(target,
119                 0,
120                 dstPixelFormat,
121                 texture.getTextureWidth(),
122                 texture.getTextureHeight(),
123                 0,
124                 srcPixelFormat,
125                 SGL.GL_UNSIGNED_BYTE,
126                 textureBuffer);
127 
128         return texture;
129     }
