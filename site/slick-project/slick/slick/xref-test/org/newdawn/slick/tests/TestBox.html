55      /**
56       * Move to the next game
57       */
58      private void nextGame() {
59          if (index == -1) {
60              return;
61          }
62  
63          index++;
64          if (index >= games.size()) {
65              index = 0;
66          }
67  
68          startGame();
69      }

71      /**
72       * Start a particular game
73       */
74      private void startGame() {
75          try {
76              currentGame = (BasicGame) ((Class) games.get(index)).newInstance();
77              container.getGraphics().setBackground(Color.black);
78              currentGame.init(container);
79              currentGame.render(container, container.getGraphics());
80          } catch (Exception e) {
81              Log.error(e);
82          }
83  
84          container.setTitle(currentGame.getTitle());
85      }

90      public void init(GameContainer c) throws SlickException {
91          if (games.size() == 0) {
92              currentGame = new BasicGame("NULL") {
93                  public void init(GameContainer container) throws SlickException {
94                  }
95  
96                  public void update(GameContainer container, int delta) throws SlickException {
97                  }
98  
99                  public void render(GameContainer container, Graphics g) throws SlickException {
100                 }
101             };
102             currentGame.init(c);
103             index = -1;
104         } else {
105             index = 0;
106             container = (AppGameContainer) c;
107             startGame();
108         }
109     }
