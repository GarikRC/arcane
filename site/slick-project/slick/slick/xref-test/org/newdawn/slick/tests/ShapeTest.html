70      public void createPoly(float x, float y) {
71          int size = 20;
72          int change = 10;
73  
74          randomShape = new Polygon();
75          // generate random polygon
76          randomShape.addPoint(0 + (int) (Math.random() * change), 0 + (int) (Math.random() * change));
77          randomShape.addPoint(size - (int) (Math.random() * change), 0 + (int) (Math.random() * change));
78          randomShape.addPoint(size - (int) (Math.random() * change), size - (int) (Math.random() * change));
79          randomShape.addPoint(0 + (int) (Math.random() * change), size - (int) (Math.random() * change));
80  
81          // center polygon
82          randomShape.setCenterX(x);
83          randomShape.setCenterY(y);
84      }

114             g.fill((Shape) shapes.get(i));

116         g.fill(randomShape);
