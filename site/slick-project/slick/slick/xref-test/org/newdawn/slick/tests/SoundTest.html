127     public void keyPressed(int key, char c) {
128         if (key == Input.KEY_ESCAPE) {
129             System.exit(0);
130         }
131         if (key == Input.KEY_SPACE) {
132             sound.play();
133         }
134         if (key == Input.KEY_B) {
135             burp.play();
136         }
137         if (key == Input.KEY_A) {
138             sound.playAt(-1, 0, 0);
139         }
140         if (key == Input.KEY_L) {
141             sound.playAt(1, 0, 0);
142         }
143         if (key == Input.KEY_RETURN) {
144             charlie.play(1.0f, 1.0f);
145         }
146         if (key == Input.KEY_P) {
147             if (music.playing()) {
148                 music.pause();
149             } else {
150                 music.resume();
151             }
152         }
153         if (key == Input.KEY_C) {
154             music.stop();
155             if (music == musica) {
156                 music = musicb;
157             } else {
158                 music = musica;
159             }
160 
161             music.loop();
162         }
163         for (int i = 0; i < 3; i++) {
164             if (key == Input.KEY_1 + i) {
165                 if (engines[i] != 0) {
166                     System.out.println("Stop " + i);
167                     SoundStore.get().stopSoundEffect(engines[i]);
168                     engines[i] = 0;
169                 } else {
170                     System.out.println("Start " + i);
171                     engines[i] = engine.playAsSoundEffect(1, 1, true);
172                 }
173             }
174         }
175 
176         if (c == '+') {
177             volume += 1;
178             setVolume();
179         }
180 
181         if (c == '-') {
182             volume -= 1;
183             setVolume();
184         }
185 
186         if (key == Input.KEY_Y) {
187             int vol = (int) (music.getVolume() * 10);
188             vol--;
189             if (vol < 0) vol = 0;
190             // set individual volume of music
191             music.setVolume(vol / 10.0f);
192         }
193         if (key == Input.KEY_X) {
194             int vol = (int) (music.getVolume() * 10);
195             vol++;
196             if (vol > 10) vol = 10;
197             // set individual volume of music
198             music.setVolume(vol / 10.0f);
199         }
200         if (key == Input.KEY_N) {
201             int vol = (int) (myContainer.getSoundVolume() * 10);
202             vol--;
203             if (vol < 0) vol = 0;
204             // set global volume of sound fx
205             myContainer.setSoundVolume(vol / 10.0f);
206         }
207         if (key == Input.KEY_M) {
208             int vol = (int) (myContainer.getSoundVolume() * 10);
209             vol++;
210             if (vol > 10) vol = 10;
211             // set global volume of sound fx
212             myContainer.setSoundVolume(vol / 10.0f);
213         }

215     }
