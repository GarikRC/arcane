100     public void keyPressed(int key, char c) {
101         if (key == Input.KEY_ESCAPE) {
102             System.exit(0);
103         }
104         if (key == Input.KEY_SPACE) {
105             sound.play();
106         }
107         if (key == Input.KEY_B) {
108             burp.play();
109         }
110         if (key == Input.KEY_A) {
111             sound.playAt(-1, 0, 0);
112         }
113         if (key == Input.KEY_L) {
114             sound.playAt(1, 0, 0);
115         }
116         if (key == Input.KEY_RETURN) {
117             charlie.play(1.0f, 1.0f);
118         }
119         if (key == Input.KEY_P) {
120             if (music.playing()) {
121                 music.pause();
122             } else {
123                 music.resume();
124             }
125         }
126         if (key == Input.KEY_C) {
127             music.stop();
128             if (music == musica) {
129                 music = musicb;
130             } else {
131                 music = musica;
132             }
133 
134             music.loop();
135         }
136         if (key == Input.KEY_E) {
137             if (engine.playing()) {
138                 engine.stop();
139             } else {
140                 engine.loop();
141             }
142         }
143 
144         if (c == '+') {
145             volume += 1;
146             setVolume();
147         }
148 
149         if (c == '-') {
150             volume -= 1;
151             setVolume();
152         }
153 
154     }
