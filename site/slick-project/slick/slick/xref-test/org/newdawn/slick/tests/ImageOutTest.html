53      public void init(GameContainer container) throws SlickException {

55  
56          try {
57              fire = ParticleIO.loadConfiguredSystem("testdata/system.xml");
58          } catch (IOException e) {
59              throw new SlickException("Failed to load particle systems", e);
60          }
61  
62          copy = new Image(400, 300);
63          String[] formats = ImageOut.getSupportedFormats();
64          message = "Formats supported: ";
65          for (int i = 0; i < formats.length; i++) {
66              message += formats[i];
67              if (i < formats.length - 1) {
68                  message += ",";
69              }
70          }
71      }

76      public void render(GameContainer container, Graphics g) {
77          g.drawString("T - TGA Snapshot", 10, 50);
78          g.drawString("J - JPG Snapshot", 10, 70);
79          g.drawString("P - PNG Snapshot", 10, 90);
80  
81          g.setDrawMode(Graphics.MODE_ADD);
82          g.drawImage(copy, 200, 300);
83          g.setDrawMode(Graphics.MODE_NORMAL);
84  
85          g.drawString(message, 10, 400);
86          g.drawRect(200, 0, 400, 300);
87          g.translate(400, 250);
88          fire.render();
89          this.g = g;
90      }

98      private void writeTo(String fname) throws SlickException {
