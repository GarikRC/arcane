97      public void render(GameContainer container, Graphics g) throws SlickException {
98  
99          // RENDERING TO AN IMAGE AND THEN DRAWING IT TO THE DISPLAY
100         // Draw graphics and text onto our graphics context from the Image target
101         gTarget.setBackground(new Color(0, 0, 0, 0));
102         gTarget.clear();
103         gTarget.rotate(200, 160, ang);
104         gTarget.setFont(testFont);
105         gTarget.fillRect(10, 10, 50, 50);
106         gTarget.drawString("HELLO WORLD", 10, 10);
107 
108         gTarget.drawImage(testImage, 100, 150);
109         gTarget.drawImage(testImage, 100, 50);
110         gTarget.drawImage(testImage, 50, 75);
111 
112         // Note we started by clearing the offscreen graphics area and then end
113         // by calling flush
114         gTarget.flush();
115 
116         g.setColor(Color.red);
117         g.fillRect(250, 50, 200, 200);
118         // The image has been updated using its graphics context, so now draw the image
119         // to the screen a few times
120         target.draw(300, 100);
121         target.draw(300, 410, 200, 150);
122         target.draw(505, 410, 100, 75);
123 
124         // Draw some text on the screen to indicate what we did and put some
125         // nice boxes around the three areas
126         g.setColor(Color.white);
127         g.drawString("Testing On Offscreen Buffer", 300, 80);
128         g.setColor(Color.green);
129         g.drawRect(300, 100, target.getWidth(), target.getHeight());
130         g.drawRect(300, 410, target.getWidth() / 2, target.getHeight() / 2);
131         g.drawRect(505, 410, target.getWidth() / 4, target.getHeight() / 4);
132 
133         // SCREEN COPY EXAMPLE
134         // Put some text and simple graphics on the screen to test copying
135         // from the screen to a target image
136         g.setColor(Color.white);
137         g.drawString("Testing Font On Back Buffer", 10, 100);
138         g.drawString("Using: " + using, 10, 580);
139         g.setColor(Color.red);
140         g.fillRect(10, 120, 200, 5);
141 
142         // Copy the screen area into a destination image
143         int xp = (int) (60 + (Math.sin(ang / 60) * 50));
144         g.copyArea(cut, xp, 50);
145 
146         // Draw the copied image to the screen and put some nice
147         // boxes around the source and the destination
148         cut.draw(30, 250);
149         g.setColor(Color.white);
150         g.drawRect(30, 250, cut.getWidth(), cut.getHeight());
151         g.setColor(Color.gray);
152         g.drawRect(xp, 50, cut.getWidth(), cut.getHeight());
153 
154         // ALTERING A LOADED IMAGE EXAMPLE
155         // Draw the image we loaded in the init method and then modified
156         // by drawing some text and simple geometry on it
157         preloaded.draw(2, 400);
158         g.setColor(Color.blue);
159         g.drawRect(2, 400, preloaded.getWidth(), preloaded.getHeight());
